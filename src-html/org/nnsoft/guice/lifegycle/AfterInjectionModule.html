019import com.google.inject.ProvisionException;
020import com.google.inject.matcher.Matcher;
021import com.google.inject.spi.InjectionListener;
022import com.google.inject.spi.TypeEncounter;
023
024import java.lang.annotation.Annotation;
025import java.lang.reflect.InvocationTargetException;
026import java.lang.reflect.Method;
027
028import static java.lang.String.format;
029
030/**
031 * Guice module to register methods to be invoked after injection is complete.
032 */
033public final class AfterInjectionModule
034        extends AbstractLifeCycleModule {
035    /**
036     * Creates a new module which register methods annotated with {@link AfterInjection} on methods
037     * in any type.
038     */
039    public AfterInjectionModule() {
040        super(AfterInjection.class);
041    }
042
043    /**
044     * Creates a new module which register methods annotated with input annotation on methods in
045     * types filtered by the input matcher.
046     *
047     * @param afterInjectionAnnotationType the <i>AfterInjection</i> annotation to be searched.
048     * @param typeMatcher the filter for injectee types.
049     */
050    public <A extends Annotation> AfterInjectionModule(final Class<A> afterInjectionAnnotationType,
051            final Matcher<Object> typeMatcher) {
052        super(afterInjectionAnnotationType, typeMatcher);
053    }
054
055    @Override
056    protected void configure() {
057        bindListener(getTypeMatcher(), new AbstractMethodTypeListener(getAnnotationType()) {
058            @Override
059            protected <I> void hear(final Method method, final TypeEncounter<I> encounter) {
060                encounter.register((InjectionListener<I>) injectee -> {
061                    try {
062                        method.invoke(injectee);
063                    } catch (final IllegalArgumentException e) {
064                        // should not happen, anyway...
065                        throw new ProvisionException(format("Method @%s %s requires arguments",
066                                getAnnotationType().getName(), method), e);
067                    } catch (final IllegalAccessException e) {
068                        throw new ProvisionException(format("Impossible to access to @%s %s on %s",
069                                getAnnotationType().getName(), method, injectee), e);
070                    } catch (final InvocationTargetException e) {
071                        throw new ProvisionException(
072                                format("An error occurred while invoking @%s %s on %s",
073                                        getAnnotationType().getName(), method, injectee),
074                                e.getTargetException());
075                    }
076                });
077            }
078        });
079    }
080}
