001package hm.binkley.dao;
002
003import org.springframework.dao.DataAccessException;
004import org.springframework.jdbc.core.JdbcTemplate;
005import org.springframework.jdbc.datasource.DataSourceTransactionManager;
006import org.springframework.jdbc.support.MetaDataAccessException;
007import org.springframework.jdbc.support.SQLExceptionTranslator;
008import org.springframework.transaction.TransactionStatus;
009import org.springframework.transaction.support.TransactionTemplate;
010
011import javax.annotation.Nonnull;
012import javax.annotation.Nullable;
013import javax.inject.Inject;
014import java.sql.DatabaseMetaData;
015import java.sql.SQLException;
016import java.util.List;
017import java.util.Optional;
018import java.util.function.Function;
019
020import static org.springframework.jdbc.support.JdbcUtils.extractDatabaseMetaData;
021
022/**
023 * A very simple DAO wrapper for using Spring transactions and JDBC template, designed for lambda
024 * use.  Example:
025 *
026 * <pre>
027 * SimpleDAO&lt;String&gt; someColumn = new SimpleDAO(transactionManager);
028 * String columnValue = someColumn.dao(
029 *        (jdbcTemplate, status) -&gt; jdbcTemplate.queryForObject("sql here", String.class);</pre>
030 *
031 * @author <a href="binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
032 */
033public class SimpleDAO {
034    private final DataSourceTransactionManager transactionManager;
035
036    /**
037     * Constructs a new {@code SimpleDAO} with the given data source <var>transactionManager</var>.
038     *
039     * @param transactionManager the transaction manager, never missing
040     */
041    @Inject
042    public SimpleDAO(@Nonnull final DataSourceTransactionManager transactionManager) {
043        this.transactionManager = transactionManager;
044    }
045
046    /**
047     * Access to the underlying transaction manager.
048     *
049     * @return the transaction manager, never missing
050     */
051    @Nonnull
052    public final DataSourceTransactionManager getTransactionManager() {
053        return transactionManager;
054    }
055
056    /**
057     * Gets JDBC URL for this transaction manager.  This uses two connections, managed within.
058     *
059     * @return the JDBC URL or {@code null} if not applicable for the data source
060     *
061     * @throws DataAccessException if JDBC fails
062     * @throws MetaDataAccessException if JDBC database metadata fails
063     */
064    @Nullable
065    public String getJdbcUrl()
066            throws DataAccessException, MetaDataAccessException {
067        return (String) extractDatabaseMetaData(transactionManager.getDataSource(),
068                DatabaseMetaData::getURL);
069    }
070
071    /**
072     * Runs the given <var>dao</var> within Spring transaction.
073     *
074     * @param dao the dao callback, never missing
075     * @param <T> the return type of the callback
076     *
077     * @return the callback result
078     */
079    public final <T> T dao(@Nonnull final Dao<T> dao) {
080        return dao.using(transactionManager);
081    }
082
083    /**
084     * Runs the given <var>dao</var> within Spring transaction, requiring 0 or 1 results.
085     *
086     * @param dao the dao callback, never missing
087     * @param wrongSize the exception factory for 2 or more results, never missing
088     * @param <T> the return type of the callback
089     *
090     * @return an optional of the return
091     *
092     * @throws DataAccessException if the results are the wrong size
093     */
094    public final <T> Optional<T> daoMaybe(@Nonnull final Dao<List<T>> dao,
095            @Nonnull final Function<Integer, DataAccessException> wrongSize) {
096        final List<T> result = dao(dao);
097        final int size = result.size();
098        switch (size) {
099        case 0:
100            return Optional.empty();
101        case 1:
102            return Optional.of(result.get(0));
103        default:
104            throw wrongSize.apply(size);
105        }
106    }
107
108    /**
109     * The functional interface for {@link #dao(Dao)}.
110     *
111     * @param <T> the callback return type
112     */
113    @FunctionalInterface
114    public interface Dao<T> {
115        /**
116         * Manages the JDBC callback, wrapping it in a Spring transaction.  The JDBC template passed
117         * to the callback shares the data source of the transaction manager, and executes within a
118         * transaction template.  Calls {@link #with(JdbcTemplate, TransactionStatus) with} to
119         * execute JDBC methods.
120         *
121         * @param transactionManager the transaction manager, never missing
122         *
123         * @return the callback result
124         *
125         * @throws DataAccessException if JDBC fails
126         */
127        default T using(@Nonnull final DataSourceTransactionManager transactionManager)
128                throws DataAccessException {
129            return new TransactionTemplate(transactionManager).execute(status -> {
130                final JdbcTemplate jdbcTemplate = new JdbcTemplate(
131                        transactionManager.getDataSource());
132                try {
133                    return with(jdbcTemplate, status);
134                } catch (final SQLException e) {
135                    throw jdbcTemplate.getExceptionTranslator().translate(task(), sql(), e);
136                }
137            });
138        }
139
140        /**
141         * Executes the callback, passing in the given <var>jdbcTemplate</var> and transaction
142         * <var>status</var>.  This is typically implemented as a lambda.
143         *
144         * Exceptions are translated into {@link DataAccessException} instances in {@link
145         * #using(DataSourceTransactionManager) using}.
146         *
147         * @param jdbcTemplate the JDBC template, never missing
148         * @param status the transaction status, never missing
149         *
150         * @return the callback result
151         *
152         * @throws SQLException if JDBC fails
153         */
154        T with(@Nonnull final JdbcTemplate jdbcTemplate, @Nonnull final TransactionStatus status)
155                throws SQLException;
156
157        /**
158         * Names the task for the Spring JDBC exception translator for more descriptive error
159         * messages.  Defaults to {@code null}.  Implementors <strong></strong> return a non-{@code
160         * null} values, relying on the default otherwise.
161         *
162         * @return the task name
163         *
164         * @see SQLExceptionTranslator#translate(String, String, SQLException)
165         */
166        default String task() {
167            return null;
168        }
169
170        /**
171         * Provides the executed SQL for the Spring JDBC exception translator for more descriptive
172         * error messages.  Defaults to {@code null}.  Implementors <strong></strong> return a
173         * non-{@code null} values, relying on the default otherwise.
174         *
175         * @return the executed SQL
176         *
177         * @see SQLExceptionTranslator#translate(String, String, SQLException)
178         */
179        default String sql() {
180            return null;
181        }
182    }
183}
