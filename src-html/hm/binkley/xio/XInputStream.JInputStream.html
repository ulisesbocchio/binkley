001package hm.binkley.xio;
002
003import javax.annotation.Nonnull;
004import java.io.Closeable;
005import java.io.IOException;
006import java.io.InputStream;
007
008/**
009 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
010 * @todo Javadoc
011 */
012public interface XInputStream
013        extends Closeable {
014    /**
015     * MAX_SKIP_BUFFER_SIZE is used to determine the maximum buffer size to use when skipping.
016     *
017     * @see JInputStream#MAX_SKIP_BUFFER_SIZE
018     */
019    int MAX_SKIP_BUFFER_SIZE = 2048;
020
021    /**
022     * Creates a new JDK {@code InputStream} forwarding all calls to this {@code XInputStream}.
023     *
024     * @return the new JDK input stream, never missing
025     */
026    @Nonnull
027    default JInputStream asInputStream() {
028        return new JInputStream(this);
029    }
030
031    /**
032     * Reads the next byte of data from the input stream. The value byte is returned as an {@code
033     * int} in the range {@code 0} to {@code 255}. If no byte is available because the end of the
034     * stream has been reached, the value {@code -1} is returned. This method blocks until input
035     * data is available, the end of the stream is detected, or an exception is thrown.
036     *
037     * @return the next byte of data, or {@code -1} if the end of the stream is reached.
038     *
039     * @throws IOException if an I/O error occurs.
040     */
041    int read()
042            throws IOException;
043
044    /**
045     * Reads some number of bytes from the input stream and stores them into the buffer array {@code
046     * b}. The number of bytes actually read is returned as an integer.  This method blocks until
047     * input data is available, end of file is detected, or an exception is thrown.
048     * <p>
049     * If the length of {@code b} is zero, then no bytes are read and {@code 0} is returned;
050     * otherwise, there is an attempt to read at least one byte. If no byte is available because the
051     * stream is at the end of the file, the value {@code -1} is returned; otherwise, at least one
052     * byte is read and stored into {@code b}.
053     * <p>
054     * The first byte read is stored into element {@code b[0]}, the next one into {@code b[1]}, and
055     * so on. The number of bytes read is, at most, equal to the length of {@code b}. Let <i>k</i>
056     * be the number of bytes actually read; these bytes will be stored in elements {@code b[0]}
057     * through {@code b[}<i>k</i>{@code -1]}, leaving elements {@code b[}<i>k</i>{@code ]} through
058     * {@code b[b.length-1]} unaffected.
059     * <p>
060     * The {@code read(b)} method for class {@code InputStream}
061     * has the same effect as: <pre>{@code read(b, 0, b.length) }</pre>
062     *
063     * @param b the buffer into which the data is read.
064     *
065     * @return the total number of bytes read into the buffer, or {@code -1} if there is no more
066     * data because the end of the stream has been reached.
067     *
068     * @throws IOException If the first byte cannot be read for any reason other than the end of the
069     * file, if the input stream has been closed, or if some other I/O error occurs.
070     * @throws NullPointerException if {@code b} is {@code null}.
071     * @see XInputStream#read(byte[], int, int)
072     */
073    default int read(@Nonnull final byte[] b)
074            throws IOException {
075        return read(b, 0, b.length);
076    }
077
078    /**
079     * Reads up to {@code len} bytes of data from the input stream into an array of bytes.  An
080     * attempt is made to read as many as {@code len} bytes, but a smaller number may be read. The
081     * number of bytes actually read is returned as an integer.
082     * <p>
083     * This method blocks until input data is available, end of file is detected, or an exception is
084     * thrown.
085     * <p>
086     * If {@code len} is zero, then no bytes are read and {@code 0} is returned; otherwise, there is
087     * an attempt to read at least one byte. If no byte is available because the stream is at end of
088     * file, the value {@code -1} is returned; otherwise, at least one byte is read and stored into
089     * {@code b}.
090     * <p>
091     * The first byte read is stored into element {@code b[off]}, the next one into {@code
092     * b[off+1]}, and so on. The number of bytes read is, at most, equal to {@code len}. Let
093     * <i>k</i> be the number of bytes actually read; these bytes will be stored in elements {@code
094     * b[off]} through {@code b[off+}<i>k</i>{@code -1]}, leaving elements {@code
095     * b[off+}<i>k</i>{@code ]} through {@code b[off+len-1]} unaffected.
096     * <p>
097     * In every case, elements {@code b[0]} through {@code b[off]} and elements {@code b[off+len]}
098     * through {@code b[b.length-1]} are unaffected.
099     * <p>
100     * The {@code read(b, off, len)} method for class {@code InputStream} simply calls the method
101     * {@code read()} repeatedly. If the first such call results in an {@code IOException}, that
102     * exception is returned from the call to the {@code read(b, off, len)} method.  If any
103     * subsequent call to {@code read()} results in a {@code IOException}, the exception is caught
104     * and treated as if it were end of file; the bytes read up to that point are stored into {@code
105     * b} and the number of bytes read before the exception occurred is returned. The default
106     * implementation of this method blocks until the requested amount of input data {@code len} has
107     * been read, end of file is detected, or an exception is thrown. Subclasses are encouraged to
108     * provide a more efficient implementation of this method.
109     *
110     * @param b the buffer into which the data is read.
111     * @param off the start offset in array {@code b} at which the data is written.
112     * @param len the maximum number of bytes to read.
113     *
114     * @return the total number of bytes read into the buffer, or {@code -1} if there is no more
115     * data because the end of the stream has been reached.
116     *
117     * @throws IOException If the first byte cannot be read for any reason other than end of file,
118     * or if the input stream has been closed, or if some other I/O error occurs.
119     * @throws NullPointerException If {@code b} is {@code null}.
120     * @throws IndexOutOfBoundsException If {@code off} is negative, {@code len} is negative, or
121     * {@code len} is greater than {@code b.length - off}
122     * @see XInputStream#read()
123     */
124    default int read(@Nonnull final byte[] b, final int off, final int len)
125            throws IOException {
126        if (0 == len)
127            return 0;
128        if (0 > off || 0 > len || len > b.length - off)
129            throw new IndexOutOfBoundsException();
130
131        int c = read();
132        if (-1 == c)
133            return -1;
134        b[off] = (byte) c;
135
136        int i = 1;
137        try {
138            for (; i < len; i++) {
139                c = read();
140                if (-1 == c)
141                    break;
142                b[off + i] = (byte) c;
143            }
144        } catch (final IOException ignored) {
145        }
146        return i;
147    }
148
149    /**
150     * Skips over and discards {@code n} bytes of data from this input stream. The {@code skip}
151     * method may, for a variety of reasons, end up skipping over some smaller number of bytes,
152     * possibly {@code 0}. This may result from any of a number of conditions; reaching end of file
153     * before {@code n} bytes have been skipped is only one possibility. The actual number of bytes
154     * skipped is returned. If {@code n} is negative, the {@code skip} method for class {@code
155     * InputStream} always returns 0, and no bytes are skipped. Subclasses may handle the negative
156     * value differently.
157     * <p>
158     * The {@code skip} method of this class creates a byte array and then repeatedly reads into it
159     * until {@code n} bytes have been read or the end of the stream has been reached. Subclasses
160     * are encouraged to provide a more efficient implementation of this method. For instance, the
161     * implementation may depend on the ability to seek.
162     *
163     * @param n the number of bytes to be skipped.
164     *
165     * @return the actual number of bytes skipped.
166     *
167     * @throws IOException if the stream does not support seek, or if some other I/O error occurs.
168     */
169    default long skip(final long n)
170            throws IOException {
171        long remaining = n;
172
173        if (0 >= n)
174            return 0;
175
176        final int size = (int) Math.min(MAX_SKIP_BUFFER_SIZE, remaining);
177        final byte[] skipBuffer = new byte[size];
178        while (0 < remaining) {
179            final int nr = read(skipBuffer, 0, (int) Math.min(size, remaining));
180            if (0 > nr)
181                break;
182            remaining -= nr;
183        }
184
185        return n - remaining;
186    }
187
188    /**
189     * Returns an estimate of the number of bytes that can be read (or skipped over) from this input
190     * stream without blocking by the next invocation of a method for this input stream. The next
191     * invocation might be the same thread or another thread.  A single read or skip of this many
192     * bytes will not block, but may read or skip fewer bytes.
193     * <p>
194     * Note that while some implementations of {@code InputStream} will return the total number of
195     * bytes in the stream, many will not.  It is never correct to use the return value of this
196     * method to allocate a buffer intended to hold all data in this stream.
197     * <p>
198     * A subclass' implementation of this method may choose to throw an {@link IOException} if this
199     * input stream has been closed by invoking the {@link #close()} method.
200     * <p>
201     * The {@code available} method for class {@code InputStream} always returns {@code 0}.
202     * <p>
203     * This method should be overridden by subclasses.
204     *
205     * @return an estimate of the number of bytes that can be read (or skipped over) from this input
206     * stream without blocking or {@code 0} when it reaches the end of the input stream.
207     *
208     * @throws IOException if an I/O error occurs.
209     */
210    default int available()
211            throws IOException {
212        return 0;
213    }
214
215    /**
216     * Closes this input stream and releases any system resources associated with the stream.
217     * <p>
218     * The {@code close} method of {@code InputStream} does nothing.
219     *
220     * @throws IOException if an I/O error occurs.
221     */
222    default void close()
223            throws IOException {
224    }
225
226    class JInputStream
227            extends InputStream {
228        private final XInputStream in;
229
230        protected JInputStream(@Nonnull final XInputStream in) {
231            this.in = in;
232        }
233
234        @Override
235        public int read()
236                throws IOException {
237            return in.read();
238        }
239
240        @Override
241        public int read(@Nonnull final byte[] b)
242                throws IOException {
243            return in.read(b);
244        }
245
246        @Override
247        public int read(@Nonnull final byte[] b, final int off, final int len)
248                throws IOException {
249            return in.read(b, off, len);
250        }
251
252        @Override
253        public long skip(final long n)
254                throws IOException {
255            return in.skip(n);
256        }
257
258        @Override
259        public int available()
260                throws IOException {
261            return in.available();
262        }
263
264        @Override
265        public void close()
266                throws IOException {
267            in.close();
268        }
269
270        @Override
271        public synchronized void mark(final int readlimit) {
272            if (markSupported())
273                ((XMarkable) in).mark(readlimit);
274        }
275
276        @Override
277        public synchronized void reset()
278                throws IOException {
279            if (markSupported())
280                ((XMarkable) in).reset();
281            else
282                throw new IOException("mark/reset not supported");
283        }
284
285        @Override
286        public boolean markSupported() {
287            return in instanceof XMarkable;
288        }
289    }
