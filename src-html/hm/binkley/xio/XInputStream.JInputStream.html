001package hm.binkley.xio;
002
003import javax.annotation.Nonnull;
004import java.io.Closeable;
005import java.io.IOException;
006import java.io.InputStream;
007
008/**
009 * {@code XInputStream} is a minimal rewrite of {@link InputStream}.
010 *
011 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
012 */
013public interface XInputStream
014        extends Closeable {
015    /**
016     * MAX_SKIP_BUFFER_SIZE is used to determine the maximum buffer size to use
017     * when skipping.
018     *
019     * @see JInputStream#MAX_SKIP_BUFFER_SIZE
020     */
021    int MAX_SKIP_BUFFER_SIZE = 2048;
022
023    /**
024     * Creates a new JDK {@code InputStream} forwarding all calls to this {@code
025     * XInputStream}.
026     *
027     * @return the new JDK input stream, never missing
028     */
029    @Nonnull
030    default JInputStream asInputStream() {
031        return new JInputStream(this);
032    }
033
034    /**
035     * Reads the next byte of data from the input stream. The value byte is
036     * returned as an {@code int} in the range {@code 0} to {@code 255}. If no
037     * byte is available because the end of the stream has been reached, the
038     * value {@code -1} is returned. This method blocks until input data is
039     * available, the end of the stream is detected, or an exception is thrown.
040     *
041     * @return the next byte of data, or {@code -1} if the end of the stream is
042     * reached.
043     *
044     * @throws IOException if an I/O error occurs.
045     */
046    int read()
047            throws IOException;
048
049    /**
050     * Reads some number of bytes from the input stream and stores them into the
051     * buffer array {@code b}. The number of bytes actually read is returned as
052     * an integer.  This method blocks until input data is available, end of
053     * file is detected, or an exception is thrown.
054     * <p>
055     * If the length of {@code b} is zero, then no bytes are read and {@code 0}
056     * is returned; otherwise, there is an attempt to read at least one byte. If
057     * no byte is available because the stream is at the end of the file, the
058     * value {@code -1} is returned; otherwise, at least one byte is read and
059     * stored into {@code b}.
060     * <p>
061     * The first byte read is stored into element {@code b[0]}, the next one
062     * into {@code b[1]}, and so on. The number of bytes read is, at most, equal
063     * to the length of {@code b}. Let <i>k</i> be the number of bytes actually
064     * read; these bytes will be stored in elements {@code b[0]} through {@code
065     * b[}<i>k</i>{@code -1]}, leaving elements {@code b[}<i>k</i>{@code ]}
066     * through {@code b[b.length-1]} unaffected.
067     * <p>
068     * The {@code read(b)} method for class {@code InputStream}
069     * has the same effect as: <pre>{@code read(b, 0, b.length) }</pre>
070     *
071     * @param b the buffer into which the data is read.
072     *
073     * @return the total number of bytes read into the buffer, or {@code -1} if
074     * there is no more data because the end of the stream has been reached.
075     *
076     * @throws IOException If the first byte cannot be read for any reason other
077     * than the end of the file, if the input stream has been closed, or if some
078     * other I/O error occurs.
079     * @throws NullPointerException if {@code b} is {@code null}.
080     * @see XInputStream#read(byte[], int, int)
081     */
082    default int read(@Nonnull final byte[] b)
083            throws IOException {
084        return read(b, 0, b.length);
085    }
086
087    /**
088     * Reads up to {@code len} bytes of data from the input stream into an array
089     * of bytes.  An attempt is made to read as many as {@code len} bytes, but a
090     * smaller number may be read. The number of bytes actually read is returned
091     * as an integer.
092     * <p>
093     * This method blocks until input data is available, end of file is
094     * detected, or an exception is thrown.
095     * <p>
096     * If {@code len} is zero, then no bytes are read and {@code 0} is returned;
097     * otherwise, there is an attempt to read at least one byte. If no byte is
098     * available because the stream is at end of file, the value {@code -1} is
099     * returned; otherwise, at least one byte is read and stored into {@code
100     * b}.
101     * <p>
102     * The first byte read is stored into element {@code b[off]}, the next one
103     * into {@code b[off+1]}, and so on. The number of bytes read is, at most,
104     * equal to {@code len}. Let <i>k</i> be the number of bytes actually read;
105     * these bytes will be stored in elements {@code b[off]} through {@code
106     * b[off+}<i>k</i>{@code -1]}, leaving elements {@code b[off+}<i>k</i>{@code
107     * ]} through {@code b[off+len-1]} unaffected.
108     * <p>
109     * In every case, elements {@code b[0]} through {@code b[off]} and elements
110     * {@code b[off+len]} through {@code b[b.length-1]} are unaffected.
111     * <p>
112     * The {@code read(b, off, len)} method for class {@code InputStream} simply
113     * calls the method {@code read()} repeatedly. If the first such call
114     * results in an {@code IOException}, that exception is returned from the
115     * call to the {@code read(b, off, len)} method.  If any subsequent call to
116     * {@code read()} results in a {@code IOException}, the exception is caught
117     * and treated as if it were end of file; the bytes read up to that point
118     * are stored into {@code b} and the number of bytes read before the
119     * exception occurred is returned. The default implementation of this method
120     * blocks until the requested amount of input data {@code len} has been
121     * read, end of file is detected, or an exception is thrown. Subclasses are
122     * encouraged to provide a more efficient implementation of this method.
123     *
124     * @param b the buffer into which the data is read.
125     * @param off the start offset in array {@code b} at which the data is
126     * written.
127     * @param len the maximum number of bytes to read.
128     *
129     * @return the total number of bytes read into the buffer, or {@code -1} if
130     * there is no more data because the end of the stream has been reached.
131     *
132     * @throws IOException If the first byte cannot be read for any reason other
133     * than end of file, or if the input stream has been closed, or if some
134     * other I/O error occurs.
135     * @throws NullPointerException If {@code b} is {@code null}.
136     * @throws IndexOutOfBoundsException If {@code off} is negative, {@code len}
137     * is negative, or {@code len} is greater than {@code b.length - off}
138     * @see XInputStream#read()
139     */
140    default int read(@Nonnull final byte[] b, final int off, final int len)
141            throws IOException {
142        if (0 == len)
143            return 0;
144        if (0 > off || 0 > len || len > b.length - off)
145            throw new IndexOutOfBoundsException();
146
147        int c = read();
148        if (-1 == c)
149            return -1;
150        b[off] = (byte) c;
151
152        int i = 1;
153        try {
154            for (; i < len; i++) {
155                c = read();
156                if (-1 == c)
157                    break;
158                b[off + i] = (byte) c;
159            }
160        } catch (final IOException ignored) {
161        }
162        return i;
163    }
164
165    /**
166     * Skips over and discards {@code n} bytes of data from this input stream.
167     * The {@code skip} method may, for a variety of reasons, end up skipping
168     * over some smaller number of bytes, possibly {@code 0}. This may result
169     * from any of a number of conditions; reaching end of file before {@code n}
170     * bytes have been skipped is only one possibility. The actual number of
171     * bytes skipped is returned. If {@code n} is negative, the {@code skip}
172     * method for class {@code InputStream} always returns 0, and no bytes are
173     * skipped. Subclasses may handle the negative value differently.
174     * <p>
175     * The {@code skip} method of this class creates a byte array and then
176     * repeatedly reads into it until {@code n} bytes have been read or the end
177     * of the stream has been reached. Subclasses are encouraged to provide a
178     * more efficient implementation of this method. For instance, the
179     * implementation may depend on the ability to seek.
180     *
181     * @param n the number of bytes to be skipped.
182     *
183     * @return the actual number of bytes skipped.
184     *
185     * @throws IOException if the stream does not support seek, or if some other
186     * I/O error occurs.
187     */
188    default long skip(final long n)
189            throws IOException {
190        long remaining = n;
191
192        if (0 >= n)
193            return 0;
194
195        final int size = (int) Math.min(MAX_SKIP_BUFFER_SIZE, remaining);
196        final byte[] skipBuffer = new byte[size];
197        while (0 < remaining) {
198            final int nr = read(skipBuffer, 0, (int) Math.min(size, remaining));
199            if (0 > nr)
200                break;
201            remaining -= nr;
202        }
203
204        return n - remaining;
205    }
206
207    /**
208     * Returns an estimate of the number of bytes that can be read (or skipped
209     * over) from this input stream without blocking by the next invocation of a
210     * method for this input stream. The next invocation might be the same
211     * thread or another thread.  A single read or skip of this many bytes will
212     * not block, but may read or skip fewer bytes.
213     * <p>
214     * Note that while some implementations of {@code InputStream} will return
215     * the total number of bytes in the stream, many will not.  It is never
216     * correct to use the return value of this method to allocate a buffer
217     * intended to hold all data in this stream.
218     * <p>
219     * A subclass' implementation of this method may choose to throw an {@link
220     * IOException} if this input stream has been closed by invoking the {@link
221     * #close()} method.
222     * <p>
223     * The {@code available} method for class {@code InputStream} always returns
224     * {@code 0}.
225     * <p>
226     * This method should be overridden by subclasses.
227     *
228     * @return an estimate of the number of bytes that can be read (or skipped
229     * over) from this input stream without blocking or {@code 0} when it
230     * reaches the end of the input stream.
231     *
232     * @throws IOException if an I/O error occurs.
233     */
234    default int available()
235            throws IOException {
236        return 0;
237    }
238
239    /**
240     * Closes this input stream and releases any system resources associated
241     * with the stream.
242     * <p>
243     * The {@code close} method of {@code InputStream} does nothing.
244     *
245     * @throws IOException if an I/O error occurs.
246     */
247    default void close()
248            throws IOException {
249    }
250
251    class JInputStream
252            extends InputStream {
253        private final XInputStream in;
254
255        protected JInputStream(@Nonnull final XInputStream in) {
256            this.in = in;
257        }
258
259        @Override
260        public int read()
261                throws IOException {
262            return in.read();
263        }
264
265        @Override
266        public int read(@Nonnull final byte[] b)
267                throws IOException {
268            return in.read(b);
269        }
270
271        @Override
272        public int read(@Nonnull final byte[] b, final int off, final int len)
273                throws IOException {
274            return in.read(b, off, len);
275        }
276
277        @Override
278        public long skip(final long n)
279                throws IOException {
280            return in.skip(n);
281        }
282
283        @Override
284        public int available()
285                throws IOException {
286            return in.available();
287        }
288
289        @Override
290        public void close()
291                throws IOException {
292            in.close();
293        }
294
295        @Override
296        public synchronized void mark(final int readlimit) {
297            if (markSupported())
298                ((XMarkable) in).mark(readlimit);
299        }
300
301        @Override
302        public synchronized void reset()
303                throws IOException {
304            if (markSupported())
305                ((XMarkable) in).reset();
306            else
307                throw new IOException("mark/reset not supported");
308        }
309
310        @Override
311        public boolean markSupported() {
312            return in instanceof XMarkable;
313        }
314    }
