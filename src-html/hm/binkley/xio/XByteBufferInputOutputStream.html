001package hm.binkley.xio;
002
003import hm.binkley.util.Bug;
004
005import javax.annotation.Nonnull;
006import java.io.IOException;
007import java.nio.ByteBuffer;
008
009/**
010 * {@code XByteArrayInputOutputStream} is a blend of {@link XInputOutputStream}
011 * and {@link XSeekable} over a byte buffer.
012 *
013 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
014 */
015public class XByteBufferInputOutputStream
016        implements XInputOutputStream, XSeekable {
017    private final ByteBuffer buf;
018    private int mark;
019
020    /**
021     * Constructs a new {@code XByteBufferInputOutputStream} for the given byte
022     * buffer.
023     *
024     * @param buf the byte buffer, never missing
025     */
026    public XByteBufferInputOutputStream(@Nonnull final ByteBuffer buf) {
027        this.buf = buf;
028    }
029
030    @Override
031    public int read() {
032        return buf.get();
033    }
034
035    @Override
036    public void write(final int b)
037            throws IOException {
038        buf.put((byte) b);
039    }
040
041    @Override
042    public void write(@Nonnull final byte[] b, final int off, final int len)
043            throws IOException {
044        buf.put(b, off, len);
045    }
046
047    @Override
048    public long mark() {
049        return mark;
050    }
051
052    @Override
053    public void mark(final long mark) {
054        this.mark = (int) mark;
055    }
056
057    @Override
058    public long seek(final long offset, @Nonnull final Whence whence)
059            throws IOException {
060        try {
061            switch (whence) {
062            case SET:
063                buf.position((int) offset);
064                break;
065            case CUR:
066                buf.position((int) offset + buf.position());
067                break;
068            case END:
069                buf.position((int) offset + buf.limit());
070                break;
071            default:
072                throw new Bug("Missing case: " + whence);
073            }
074            return buf.position();
075        } catch (final IllegalArgumentException e) {
076            throw new IOException(e);
077        }
078    }
079}
