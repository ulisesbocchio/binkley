001package hm.binkley.xio;
002
003import java.io.IOException;
004
005/**
006 * {@code Markable} represents markable I/O streams: those which may mark a position and return to
007 * it later.
008 *
009 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
010 */
011public interface XMarkable {
012    /**
013     * Gets the most recent marked position or -1 if none.
014     *
015     * @return the most recent marked position or -1
016     *
017     * @see #mark(long)
018     */
019    long mark();
020
021    /**
022     * Marks the current position in this input stream. A subsequent call to the {@code reset}
023     * method repositions this stream at the last marked position so that subsequent reads re-read
024     * the same bytes.
025     *
026     * <p> The {@code mark} arguments tells this input stream to allow that many bytes to be
027     * read before the mark position gets invalidated.
028     *
029     * <p> The general contract of {@code mark} is that, if the method
030     * {@code markSupported} returns {@code true}, the stream somehow remembers all the
031     * bytes read after the call to {@code mark} and stands ready to supply those same bytes
032     * again if and whenever the method {@code reset} is called.  However, the stream is not
033     * required to remember any data at all if more than {@code mark} bytes are read from the
034     * stream before {@code reset} is called.
035     *
036     * <p> Marking a closed stream should not have any effect on the stream.
037     *
038     * <p> The {@code mark} method of {@code InputStream} does nothing.
039     *
040     * @param mark the maximum limit of bytes that can be read before the mark position becomes
041     * invalid.
042     *
043     * @see #reset()
044     */
045    void mark(final long mark);
046
047    /**
048     * Repositions this stream to the position at the time the {@code mark} method was last
049     * called on this input stream.
050     *
051     * <p> The general contract of {@code reset} is:
052     *
053     * <ul> <li> If the method {@code markSupported} returns {@code true}, then:
054     *
055     * <ul><li> If the method {@code mark} has not been called since the stream was created, or
056     * the number of bytes read from the stream since {@code mark} was last called is larger
057     * than the argument to {@code mark} at that last call, then an {@code IOException}
058     * might be thrown.
059     *
060     * <li> If such an {@code IOException} is not thrown, then the stream is reset to a state
061     * such that all the bytes read since the most recent call to {@code mark} (or since the
062     * start of the file, if {@code mark} has not been called) will be resupplied to subsequent
063     * callers of the {@code read} method, followed by any bytes that otherwise would have been
064     * the next input data as of the time of the call to {@code reset}. </ul>
065     *
066     * <li> If the method {@code markSupported} returns {@code false}, then:
067     *
068     * <ul><li> The call to {@code reset} may throw an {@code IOException}.
069     *
070     * <li> If an {@code IOException} is not thrown, then the stream is reset to a fixed state
071     * that depends on the particular type of the input stream and how it was created. The bytes
072     * that will be supplied to subsequent callers of the {@code read} method depend on the
073     * particular type of the input stream. </ul></ul>
074     *
075     * <p>The method {@code reset} for class {@code InputStream} does nothing except throw
076     * an {@code IOException}.
077     *
078     * @throws IOException if this stream has not been marked or if the mark has been invalidated.
079     * @see #mark(long)
080     * @see IOException
081     */
082    void reset()
083            throws IOException;
084}
