001package hm.binkley.xio;
002
003import javax.annotation.Nonnull;
004import java.io.Closeable;
005import java.io.Flushable;
006import java.io.IOException;
007import java.io.OutputStream;
008
009/**
010 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
011 * @todo Javadoc
012 */
013public interface XOutputStream
014        extends Closeable, Flushable {
015    /**
016     * Creates a JDK {@code OutputStream} implementation forwarding all calls to this {@code
017     * XOutputStream}.
018     *
019     * @return the new JDK output stream, never missing
020     */
021    @Nonnull
022    default JOutputStream asOutputStream() {
023        return new JOutputStream(this);
024    }
025
026    /**
027     * Writes the specified byte to this output stream. The general contract for {@code write} is
028     * that one byte is written to the output stream. The byte to be written is the eight low-order
029     * bits of the argument {@code b}. The 24 high-order bits of {@code b} are ignored. <p>
030     * Subclasses of {@code OutputStream} must provide an implementation for this method.
031     *
032     * @param b the {@code byte}.
033     *
034     * @throws IOException if an I/O error occurs. In particular, an {@code IOException} may be
035     * thrown if the output stream has been closed.
036     */
037    void write(final int b)
038            throws IOException;
039
040    /**
041     * Writes {@code b.length} bytes from the specified byte array to this output stream. The
042     * general contract for {@code write(b)} is that it should have exactly the same effect as the
043     * call {@code write(b, 0, b.length)}.
044     *
045     * @param b the data.
046     *
047     * @throws IOException if an I/O error occurs.
048     * @see XOutputStream#write(byte[], int, int)
049     */
050    default void write(@Nonnull final byte[] b)
051            throws IOException {
052        write(b, 0, b.length);
053    }
054
055    /**
056     * Writes {@code len} bytes from the specified byte array starting at offset {@code off} to this
057     * output stream. The general contract for {@code write(b, off, len)} is that some of the bytes
058     * in the array {@code b} are written to the output stream in order; element {@code b[off]} is
059     * the first byte written and {@code b[off+len-1]} is the last byte written by this operation.
060     * <p> The {@code write} method of {@code OutputStream} calls the write method of one argument
061     * on each of the bytes to be written out. Subclasses are encouraged to override this method and
062     * provide a more efficient implementation. <p> If {@code b} is {@code null}, a {@code
063     * NullPointerException} is thrown. <p> If {@code off} is negative, or {@code len} is negative,
064     * or {@code off+len} is greater than the length of the array {@code b}, then an
065     * <tt>IndexOutOfBoundsException</tt> is thrown.
066     *
067     * @param b the data.
068     * @param off the start offset in the data.
069     * @param len the number of bytes to write.
070     *
071     * @throws IOException if an I/O error occurs. In particular, an {@code IOException} is thrown
072     * if the output stream is closed.
073     */
074    default void write(@Nonnull final byte[] b, final int off, final int len)
075            throws IOException {
076        if (0 == len)
077            return;
078        if (0 > off || off > b.length || 0 > len || off + len > b.length || 0 > off + len)
079            throw new IndexOutOfBoundsException();
080        for (int i = 0; i < len; i++)
081            write(b[off + i]);
082    }
083
084    /**
085     * Flushes this output stream and forces any buffered output bytes to be written out. The
086     * general contract of {@code flush} is that calling it is an indication that, if any bytes
087     * previously written have been buffered by the implementation of the output stream, such bytes
088     * should immediately be written to their intended destination. <p> If the intended destination
089     * of this stream is an abstraction provided by the underlying operating system, for example a
090     * file, then flushing the stream guarantees only that bytes previously written to the stream
091     * are passed to the operating system for writing; it does not guarantee that they are actually
092     * written to a physical device such as a disk drive. <p> The {@code flush} method of {@code
093     * OutputStream} does nothing.
094     *
095     * @throws IOException if an I/O error occurs.
096     */
097    default void flush()
098            throws IOException {
099    }
100
101    /**
102     * Closes this output stream and releases any system resources associated with this stream. The
103     * general contract of {@code close} is that it closes the output stream. A closed stream cannot
104     * perform output operations and cannot be reopened. <p> The {@code close} method of {@code
105     * OutputStream} does nothing.
106     *
107     * @throws IOException if an I/O error occurs.
108     */
109    default void close()
110            throws IOException {
111    }
112
113    class JOutputStream
114            extends OutputStream {
115        private final XOutputStream out;
116
117        protected JOutputStream(final XOutputStream out) {
118            this.out = out;
119        }
120
121        @Override
122        public void write(final int b)
123                throws IOException {
124            out.write(b);
125        }
126
127        @Override
128        public void write(@Nonnull final byte[] b)
129                throws IOException {
130            out.write(b);
131        }
132
133        @Override
134        public void write(@Nonnull final byte[] b, final int off, final int len)
135                throws IOException {
136            out.write(b, off, len);
137        }
138
139        @Override
140        public void flush()
141                throws IOException {
142            out.flush();
143        }
144
145        @Override
146        public void close()
147                throws IOException {
148            out.close();
149        }
150    }
151}
