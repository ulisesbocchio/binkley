007package hm.binkley.inject;
008
009import com.google.inject.AbstractModule;
010import com.google.inject.Module;
011import org.kohsuke.MetaInfServices;
012
013import static java.util.ServiceLoader.load;
014
015/**
016 * {@code ServicesModule} installs Guice modules annotated with {@link MetaInfServices} on Guice
017 * {@link Module} via the JDK service loader.  This is a convenient way to bootstrap Guice modules
018 * by dropping {@code META-INF/services/com.google.inject.Module} files into the classpath.
019 *
020 * Example: <pre>
021 * &#64;MetaInfServices(Module.class)
022 * public final class FooModule
023 *         extends AbstractModule {
024 *     &#64;Override
025 *     protected void configure() {
026 *         // Bindings here
027 *     }
028 * }</pre> At compile time the annotation processor for {@code MetaInfServces} writes "FooModule"
029 * into {@code META-INF/services/com.google.inject.Module}.  At runtime this module finds {@code
030 * FooModule} with {@code ServiceLoader.load(Module.class)} and installs in.
031 *
032 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
033 */
034public final class ServicesModule
035        extends AbstractModule {
036    @Override
037    protected void configure() {
038        for (final Module module : load(Module.class))
039            install(module);
040    }
041}
