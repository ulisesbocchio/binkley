007package hm.binkley.inject;
008
009import javax.annotation.Nonnull;
010
011/**
012 * {@code UncheckedThrow} throws checked exceptions as unchecked.
013 *
014 * @author <a href="mailto:roman.stoffel@gamlor.info">Roman Stoffel (Gamlor)</a>
015 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
016 * @see <a href="http://www.gamlor.info/wordpress/2010/02/throwing-checked-excpetions-like-unchecked-exceptions-in-java/">Throwing
017 * Checked Exceptions Like Unchecked Exceptions in Java</a>
018 */
019public final class UncheckedThrow {
020    UncheckedThrow() {}
021
022    /**
023     * Throws a checked exception as unchecked
024     *
025     * @param e the exception to rethrow, never missing
026     * @param <T> the return type when syntactically required, never used
027     *
028     * @return ignored
029     */
030    public static <T> T throwUnchecked(@Nonnull final Exception e) {
031        // Now we use the 'generic' method. Normally the type T is inferred
032        // from the parameters. However you can specify the type also explicit!
033        // Now we du just that! We use the RuntimeException as type!
034        // That means the throwsUnchecked throws an unchecked exception!
035        // Since the types are erased, no type-information is there to prevent this!
036        UncheckedThrow.<RuntimeException>throwsUnchecked(e);
037
038        // This is here is only to satisfy the compiler. It's actually unreachable code!
039        throw new AssertionError(
040                "This code should be unreachable. Something went terrible wrong here!");
041    }
042
043    /**
044     * Remember, Generics are erased in Java. So this basically throws an Exception. The real Type
045     * of T is lost during the compilation
046     */
047    private static <T extends Exception> void throwsUnchecked(final Exception toThrow)
048            throws T {
049        // Since the type is erased, this cast actually does nothing!!!
050        // we can throw any exception
051        throw (T) toThrow;
052    }
053}
