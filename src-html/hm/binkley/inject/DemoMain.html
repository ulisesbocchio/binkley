007package hm.binkley.inject;
008
009import com.google.inject.Inject;
010import hm.binkley.inject.DemoMain.DemoMainConfig;
011import joptsimple.OptionDeclarer;
012import org.aeonbits.owner.Config;
013import org.kohsuke.MetaInfServices;
014
015import javax.annotation.Nonnull;
016import javax.annotation.PostConstruct;
017import java.util.ServiceLoader;
018
019import static java.lang.String.format;
020import static java.lang.System.out;
021
022/**
023 * {@code DemoMain} needs documentation.
024 *
025 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
026 * @version $Id: $Id
027 * @todo Needs documentation.
028 */
029@MetaInfServices
030public final class DemoMain
031        extends Main<DemoMainConfig> {
032    private DemoMainConfig config;
033
034    /**
035     * Constructs a new {@code DemoMain}.  Note the use of {@link DemoMainConfig} with {@code
036     * super}.  This does <em>not</em> use constructor injection so that {@link ServiceLoader} may
037     * instantiate freely.
038     */
039    public DemoMain() {
040        super(DemoMainConfig.class, declarer -> declarer.accepts("debug"));
041    }
042
043    /**
044     * Injects {@link #config} with the given <var>config</var>.
045     *
046     * @param config the config, never missing
047     */
048    @Inject
049    public void setConfig(@Nonnull final DemoMainConfig config) { this.config = config; }
050
051    @Override
052    protected void addOptions(@Nonnull final OptionDeclarer optionDeclarer) {
053        optionDeclarer.accepts("debug");
054    }
055
056    /** For demonstration. */
057    @PostConstruct
058    public void init() {
059        out.println(format("DemoMain.init = %s", config.debug()));
060    }
061
062    public interface DemoMainConfig
063            extends Config {
064        @DefaultValue("false")
065        boolean debug();
066    }
067}
