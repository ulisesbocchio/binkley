007package hm.binkley.inject;
008
009import java.lang.annotation.Documented;
010import java.lang.annotation.Retention;
011import java.lang.annotation.Target;
012
013import static java.lang.annotation.ElementType.METHOD;
014import static java.lang.annotation.ElementType.TYPE;
015import static java.lang.annotation.RetentionPolicy.RUNTIME;
016
017/**
018 * Marks a class or method for entry/exit tracing, if assertions enabled for it.
019 *
020 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
021 */
022@Documented
023@Target({TYPE, METHOD})
024@Retention(RUNTIME)
025public @interface Trace {
026    /**
027     * Traces regardless of desired assertion status, default {@code false}.
028     *
029     * @return {@code true} trace regardless of {@code -ea} JVM flag
030     */
031    boolean ignoreAssertions() default false;
032}
