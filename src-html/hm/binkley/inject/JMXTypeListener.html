007package hm.binkley.inject;
008
009import com.google.inject.Binder;
010import com.google.inject.TypeLiteral;
011import com.google.inject.matcher.AbstractMatcher;
012import com.google.inject.spi.InjectionListener;
013import com.google.inject.spi.TypeEncounter;
014import com.google.inject.spi.TypeListener;
015import org.softee.management.annotation.MBean;
016import org.softee.management.exception.ManagementException;
017import org.softee.management.helper.MBeanRegistration;
018
019import javax.annotation.Nonnull;
020import javax.management.MalformedObjectNameException;
021
022import static com.google.common.base.Throwables.getRootCause;
023
024/**
025 * {@code JMXTypeListener} registers JMX mbeans using pojo-mbean.
026 *
027 * MBean registration errors are rethrown as {@code RuntimeException} to get Guice to correctly
028 * handle them as provision exceptions.  The message and call stack us updated to reflect the actual
029 * exception.
030 *
031 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
032 */
033public final class JMXTypeListener
034        implements TypeListener {
035    private static final MBeanMatcher MATCHER = new MBeanMatcher();
036    private static final JMXTypeListener LISTENER = new JMXTypeListener();
037
038    /**
039     * Binds a {@code JMXTypeListener} to the given <var>binder</var>, matching on types annotated
040     * with pojo-mbean {@link MBean}.
041     *
042     * @param binder the Guice binder, never missing
043     */
044    public static void bindJMX(@Nonnull final Binder binder) {
045        binder.bindListener(MATCHER, LISTENER);
046    }
047
048    @Override
049    public <I> void hear(final TypeLiteral<I> type, final TypeEncounter<I> encounter) {
050        encounter.register((InjectionListener<I>) injectee -> {
051            try {
052                new MBeanRegistration(injectee).register();
053            } catch (final MalformedObjectNameException | ManagementException e) {
054                // Guice will handle RuntimeException during provision.  Update the stack trace
055                // to bring the root cause to the fore
056                @SuppressWarnings("ThrowableResultOfMethodCallIgnored")
057                final Throwable cause = getRootCause(e);
058                final RuntimeException x = new RuntimeException(cause.toString());
059                x.setStackTrace(cause.getStackTrace());
060                throw x;
061            }
062        });
063    }
064
065    public static class MBeanMatcher
066            extends AbstractMatcher<TypeLiteral<?>> {
067        @Override
068        public boolean matches(final TypeLiteral<?> literal) {
069            return literal.getRawType().isAnnotationPresent(MBean.class);
070        }
071    }
072}
