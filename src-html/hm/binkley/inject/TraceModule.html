007package hm.binkley.inject;
008
009import com.google.inject.AbstractModule;
010import com.google.inject.Module;
011import com.google.inject.matcher.AbstractMatcher;
012import org.aopalliance.intercept.MethodInterceptor;
013import org.aopalliance.intercept.MethodInvocation;
014import org.kohsuke.MetaInfServices;
015import org.slf4j.ext.XLogger;
016
017import java.lang.reflect.AnnotatedElement;
018import java.lang.reflect.Field;
019import java.lang.reflect.Method;
020
021import static com.google.inject.matcher.Matchers.any;
022import static org.slf4j.ext.XLoggerFactory.getXLogger;
023
024/**
025 * {@code TraceModule} is a Guice module to enable tracing via SLF4J of method entry/exit and
026 * exceptions thrown (exception {@code Error}s).  Use {@link ServicesModule} to automatically
027 * enable. Use {@link Trace @Trace} to mark classes or individual methods for tracing. Use <a
028 * href="http://docs.oracle.com/javase/7/docs/technotes/guides/language/assert.html#enable-disable">
029 * {@code -ea}</a> to enable/disable at runtime.
030 *
031 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
032 */
033@MetaInfServices(Module.class)
034public final class TraceModule
035        extends AbstractModule {
036    @Override
037    protected void configure() {
038        final ShouldTrace matcher = new ShouldTrace();
039        final Tracer interceptor = new Tracer();
040        bindInterceptor(matcher, any(), interceptor);
041        bindInterceptor(any(), matcher, interceptor);
042    }
043
044    private static final class Tracer
045            implements MethodInterceptor {
046        private static XLogger logger(final MethodInvocation invocation)
047                throws IllegalAccessException {
048            final Class<?> declaring = invocation.getMethod().getDeclaringClass();
049            for (final Field field : declaring.getDeclaredFields())
050                if (XLogger.class.isAssignableFrom(field.getType())) {
051                    field.setAccessible(true);
052                    return XLogger.class.cast(field.get(invocation.getThis()));
053                }
054            return getXLogger(declaring);
055        }
056
057        @Override
058        public Object invoke(final MethodInvocation invocation)
059                throws Throwable {
060            final XLogger logger = logger(invocation);
061            logger.entry(invocation.getArguments());
062            try {
063                final Object proceed = invocation.proceed();
064                if (void.class == invocation.getMethod().getReturnType())
065                    logger.exit();
066                else
067                    logger.exit(proceed);
068                return proceed;
069            } catch (final Error e) {
070                // Pass Error straight through, no logging - state of VM unknown
071                throw e;
072            } catch (final Throwable e) {
073                throw logger.throwing(e);
074            }
075        }
076    }
077
078    private static final class ShouldTrace
079            extends AbstractMatcher<AnnotatedElement> {
080        @Override
081        public boolean matches(final AnnotatedElement element) {
082            final Class<?> type;
083            if (element instanceof Method)
084                type = Method.class.cast(element).getDeclaringClass();
085            else if (element instanceof Class)
086                type = Class.class.cast(element);
087            else
088                return false;
089
090            final Trace trace = element.getAnnotation(Trace.class);
091            return null != trace && (trace.ignoreAssertions() || type.desiredAssertionStatus());
092        }
093    }
094}
