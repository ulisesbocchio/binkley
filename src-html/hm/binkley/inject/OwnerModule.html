007package hm.binkley.inject;
008
009import com.google.inject.AbstractModule;
010import org.aeonbits.owner.Config;
011import org.aeonbits.owner.ConfigFactory;
012
013import javax.annotation.Nonnull;
014import java.util.HashMap;
015import java.util.Map;
016
017/**
018 * {@code OwnerModule} is light-weight wiring of an OWNER API {@link Config config} instance into
019 * Guice.
020 *
021 * Guice does not support generic provider methods; each config needs to be separately configured.
022 *
023 * @param <C> the config type
024 *
025 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
026 */
027public final class OwnerModule<C extends Config>
028        extends AbstractModule {
029    private final Class<C> configType;
030    private final Map<String, String> overrides = new HashMap<>();
031
032    /**
033     * Creates a new Guice module for the given OWNER API config class.
034     *
035     * @param configType the config class, never missing
036     * @param <C> the config type
037     *
038     * @return the Guice module, never missing
039     */
040    @Nonnull
041    public static <C extends Config> OwnerModule<C> ownerModule(
042            @Nonnull final Class<C> configType) {
043        return new OwnerModule<>(configType);
044    }
045
046    /**
047     * Creates a new Guice module for the given OWNER API config class with <var>overrides</var>,
048     * for example from commandline options.
049     *
050     * @param configType the config class, never missing
051     * @param overrides more configuration, never missing
052     * @param <C> the config type
053     *
054     * @return the Guice module, never missing
055     */
056    @Nonnull
057    public static <C extends Config> OwnerModule<C> ownerModule(@Nonnull final Class<C> configType,
058            @Nonnull final Map<String, String> overrides) {
059        return new OwnerModule<>(configType, overrides);
060    }
061
062    private OwnerModule(final Class<C> configType) {
063        this.configType = configType;
064    }
065
066    private OwnerModule(final Class<C> configType, final Map<String, String> overrides) {
067        this.configType = configType;
068        this.overrides.putAll(overrides);
069    }
070
071    @Override
072    protected void configure() {
073        bind(configType).toInstance(ConfigFactory
074                .create(configType, overrides, System.getProperties(), System.getenv()));
075    }
076}
