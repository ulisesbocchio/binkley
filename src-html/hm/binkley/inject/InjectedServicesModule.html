007package hm.binkley.inject;
008
009import com.google.inject.AbstractModule;
010import com.google.inject.Injector;
011import com.google.inject.Module;
012import org.kohsuke.MetaInfServices;
013
014import javax.annotation.Nonnull;
015
016import static java.util.ServiceLoader.load;
017
018/**
019 * {@code InjectedServicesModule} installs injected Guice modules annotated with {@link
020 * MetaInfServices} on Guice {@link Module} via the JDK service loader.  This is a convenient way to
021 * bootstrap Guice modules by dropping {@code META-INF/services/com.google.inject.Module} files into
022 * the classpath.
023 *
024 * Example: <pre>
025 * &#64;MetaInfServices(Module.class)
026 * public final class FooModule
027 *         extends AbstractModule {
028 *     private final String something;
029 *
030 *     &#64;Inject
031 *     public FooModule(&#64;Named("some-property") &#64;Nonnull final String something) {
032 *         this.something = something;
033 *     }
034 *
035 *     &#64;Override
036 *     protected void configure() {
037 *         // Bindings here - use "something" field
038 *     }
039 * }</pre> At compile time the annotation processor for {@code MetaInfServces} writes "FooModule"
040 * into {@code META-INF/services/com.google.inject.Module}.  At runtime this module finds {@code
041 * FooModule} with {@code ServiceLoader.load(Module.class)}, injects it with the parent, and
042 * installs in.
043 *
044 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
045 */
046public final class InjectedServicesModule
047        extends AbstractModule {
048    private final Injector parent;
049
050    public InjectedServicesModule(@Nonnull final Injector parent) {
051        this.parent = parent;
052    }
053
054    @Override
055    protected void configure() {
056        for (final Module module : load(Module.class)) {
057            parent.injectMembers(module);
058            install(module);
059        }
060    }
061}
