025import static com.google.common.base.Joiner.on;
026import static com.google.common.collect.Iterables.getOnlyElement;
027import static com.google.inject.Guice.createInjector;
028import static hm.binkley.inject.JOptSimpleModule.bindArgs;
029import static hm.binkley.inject.JOptSimpleModule.bootstrapInjector;
030import static hm.binkley.inject.OwnerModule.ownerModule;

033/**
034 * {@code Main} is a sample guing together the Guice modules for a trivial application bootstrap.
035 * Applications extends {@code Main} and specify {@linkplain #addOptions(OptionDeclarer) command
036 * line options} with the template pattern.
037 *
038 * For a full-featured library see <a href="https://github.com/Netflix/governator/wiki">Governator</a>.
039 *
040 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
041 * @todo Rework with {@link ServiceBinder}
042 * @todo Inheritance is wrong here, try another approach
043 */
044public abstract class Main<C extends Config> {
045    private final Class<C> configType;
046    private final String prefix;
047    private final DeclareOptions declare;
048
049    /**
050     * Main entry point called by the JVM. Order of operation: <ol> <li>Find the
051     * <strong>only</strong> implementation of {@code Main} via JDK service loader.</li> <li>Create
052     * a bootstrap injector to manage command line options with {@linkplain JOptSimpleModule
053     * jopt-simple}.</li> <li>Create the application injector with {@linkplain OwnerModule OWNER
054     * API} for configuration.</li> <li>Include {@linkplain LifecycleModule lifecycle support}.</li>
055     * <li>Scan for {@linkplain ServicesModule annotated application modules}.</li> <li>Inject the
056     * instance of {@code Main} triggering lifecycle events, if any.</li> </ol>
057     *
058     * @param args the command line arguments
059     */
060    public static void main(final String... args) {
061        final Injector stage1 = bootstrapInjector();
062        bindArgs(stage1, args);
063        final Main<Config> main = getOnlyElement(ServiceLoader.load(Main.class));
064        final Injector stage2 = createInjector(
065                main.declare(stage1.getInstance(JOptSimpleModule.class)));
066        final OwnerModule<Config> ownerModule = ownerModule(main.configType,
067                mapOf(stage2.getInstance(OptionSet.class), main.prefix));
068        stage2.createChildInjector(ownerModule, new LifecycleModule(),
069                new InjectedServicesModule(stage1)).injectMembers(main);
070    }
071
072    public interface DeclareOptions {
073        void declare(@Nonnull final OptionDeclarer declarer);
074    }
075
076    /**
077     * Constructs a new {@code Main}. Use this form when mappings of command line options are
078     * one-to-one with properties configuration.
079     *
080     * @param configType the OWNER API config type, never missing
081     * @param declare declare options, if any
082     */
083    protected Main(@Nonnull final Class<C> configType, @Nullable final DeclareOptions declare) {
084        this(configType, null, declare);
085    }
086
087    /**
088     * Constructs a new {code Main}. Use this form when command line options need a
089     * <var>prefix</var> to match properties configuration, e.g., "debug" on the command line is
090     * "my.app.debug" in properties.
091     *
092     * @param configType the OWNER API config type, never missing
093     * @param prefix the optional prefix for command line options
094     * @param declare declares options, if any
095     */
096    protected Main(@Nonnull final Class<C> configType, @Nullable final String prefix,
097            @Nullable final DeclareOptions declare) {
098        this.configType = configType;
099        this.prefix = prefix;
100        this.declare = declare;
101    }
102
103    private static Map<String, String> mapOf(final OptionSet options, final String prefix) {
104        final Map<String, String> map = new HashMap<>();
105        for (final Entry<OptionSpec<?>, List<?>> entry : options.asMap().entrySet()) {
106            final OptionSpec<?> spec = entry.getKey();
107            map.put(asPropertyKey(prefix, spec),
108                    asPropertValue(entry.getValue(), options.has(spec)));
109        }
110        return unmodifiableMap(map);
111    }
112
113    private static String asPropertyKey(final String prefix, final OptionSpec<?> spec) {
114        final Collection<String> flags = spec.options();
115        for (final String flag : flags)
116            if (1 < flag.length())
117                return null == prefix ? flag : prefix + '.' + flag;
118        throw new IllegalArgumentException("No usable non-short flag: " + flags);
119    }
120
121    private <T extends OptionDeclarer> T declare(final T declarer) {
122        if (null != declare)
123            declare.declare(declarer);
124        return declarer;
125    }
126
127    private static String asPropertValue(final List<?> values, final boolean present) {
128        return values.isEmpty() ? String.valueOf(present) : on(',').join(values);
129    }
130
131    /**
132     * Configure command line parsing with JOpt-Simple.
133     *
134     * @param optionDeclarer the options parser, never mising
135     */
136    protected abstract void addOptions(@Nonnull final OptionDeclarer optionDeclarer);
137}
