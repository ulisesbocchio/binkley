007package hm.binkley.inject;
008
009import com.google.inject.AbstractModule;
010import com.google.inject.Injector;
011import org.nnsoft.guice.lifegycle.AfterInjectionModule;
012import org.nnsoft.guice.lifegycle.DisposeModule;
013import org.nnsoft.guice.lifegycle.Disposer;
014
015import javax.annotation.Nonnull;
016import javax.annotation.PostConstruct;
017import javax.annotation.PreDestroy;
018
019import static com.google.inject.matcher.Matchers.any;
020import static java.lang.Runtime.getRuntime;
021
022/**
023 * {@code LifecycleModule} enables <a href="https://en.wikipedia.org/wiki/JSR_250">the JSR-250
024 * annotations</a>, &#64;{@link PostConstruct} and &#64;{@link PreDestroy}, in Guice via the <a
025 * href="http://99soft.github.io/lifegycle/">lifegycle library</a>.  Google shows several other,
026 * more fully featured soutions; this is the smallest implementation.
027 *
028 * Guice lacks shutdown hooks.  For {@code @PreDestroy} to work you must {@link
029 * #enablePreDestroy(Injector) install one yourself}: <pre>
030 * Injector guice = Guice.createInjector(new LifecycleModule());
031 * LifecycleModule.enablePreDestroy(guice);</pre>
032 *
033 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
034 * @todo References to alternative approaches
035 * @todo Work out relation to {@code ProvisionListener} in Guice 4
036 * @todo Fix bug with listener when JDK proxy passed in (null package)
037 * @todo Work out "right thing" for installing manually v. meta-inf module
038 */
039//@MetaInfServices(Module.class)
040public final class LifecycleModule
041        extends AbstractModule {
042    /**
043     * After creating the Guice injector, invoke this method on the injector to enable {@link
044     * PreDestroy} when the JVM shuts down.
045     *
046     * @param guice the Guice injector, never missing
047     *
048     * @return the <var>guice</var> injector
049     */
050    @SuppressWarnings("UnusedDeclaration")
051    @Nonnull
052    public static Injector enablePreDestroy(@Nonnull final Injector guice) {
053        getRuntime().addShutdownHook(new Thread() {
054            @Override
055            public void run() {
056                guice.getInstance(Disposer.class).dispose();
057            }
058        });
059        return guice;
060    }
061
062    @Override
063    protected void configure() {
064        install(new AfterInjectionModule(PostConstruct.class, any()));
065        install(new DisposeModule(PreDestroy.class, any()));
066    }
067}
