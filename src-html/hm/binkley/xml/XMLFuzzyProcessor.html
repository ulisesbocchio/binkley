073/**
074 * {@code FuzzyProcessor} <b>needs documentation</b>.
075 *
076 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
077 * @todo Needs documentation.
078 * @todo Think through use of FreeMarker - logging dependencies, runtime dependency
079 * @todo Support inner interfaces
080 * @todo How to map evaluate parameters onto xpath resolver map?
081 */
082@MetaInfServices(Processor.class)
083@SupportedAnnotationTypes({"hm.binkley.xml.XMLFuzzy", "hm.binkley.xml.XMLFuzzy.Field"})
084@SupportedSourceVersion(RELEASE_8)
085@NotThreadSafe
086public class XMLFuzzyProcessor
087        extends AbstractProcessor {
088    private static final XPathFactory xpathFactory = XPathFactory.newInstance();
089    private static final Map<String, XPathExpression> expressions = new HashMap<>();
090
091    /**
092     * Evaluate an XPath <var>expression</var> against <var>node</var> using a common, shared cache
093     * for compiled expressions.
094     *
095     * @param node the document node, never missing
096     * @param expression the xpath expression, never missing
097     *
098     * @return the evaluation result
099     *
100     * @throws XPathExpressionException
101     * @todo Thread safety
102     */
103    @Nonnull
104    public static String evaluate(@Nonnull final Node node,
105            @Nonnull @Language("XPath") final String expression)
106            throws XPathExpressionException {
107        try {
108            return expressions.computeIfAbsent(expression, expr -> {
109                try {
110                    return xpathFactory.newXPath().compile(expr);
111                } catch (final XPathExpressionException e) {
112                    throw new Passing(e);
113                }
114            }).evaluate(node).intern();
115        } catch (final Passing passing) {
116            throw (XPathExpressionException) passing.getCause();
117        }
118    }
119
120    private static final class Passing
121            extends RuntimeException {
122        private Passing(final XPathExpressionException real) {
123            super(real);
124        }
125    }
126
127    @Override
128    public boolean process(final Set<? extends TypeElement> annotations,
129            final RoundEnvironment roundEnv) {
130        final Filer filer = processingEnv.getFiler();
131        final Configuration configuration = new Configuration();
132        try {
133            configuration.setDirectoryForTemplateLoading(new File(
134                    filer.getResource(CLASS_PATH, getClass().getPackage().getName(), "fuzzy.ftl")
135                            .toUri()).getParentFile());
136        } catch (final IOException e) {
137            processingEnv.getMessager().printMessage(ERROR, e.toString());
138            return false;
139        }
140
141        for (final Element element : roundEnv.getElementsAnnotatedWith(XMLFuzzy.class)) {
142            try {
143                final PackageElement packaj = processingEnv.getElementUtils().getPackageOf(element);
144                final Name simpleName = element.getSimpleName();
145                try (final Writer out = filer
146                        .createSourceFile(packaj + "." + simpleName + "Factory", element)
147                        .openWriter()) {
148                    final Map<String, Object> model = new HashMap<>();
149                    model.put("generator", XMLFuzzyProcessor.class.getName());
150                    model.put("date", Instant.now());
151                    model.put("package", packaj);
152                    model.put("simpleName", simpleName);
153                    //noinspection unchecked
154                    final Set<ExecutableElement> methodElements = (Set<ExecutableElement>) roundEnv
155                            .getElementsAnnotatedWith(Field.class);
156                    final List<Map<String, Object>> methodModels = new ArrayList<>(
157                            methodElements.size());
158                    for (final ExecutableElement methodElement : methodElements) {
159                        if (!methodElement.getEnclosingElement().equals(element))
160                            continue;
161                        if (!methodElement.getParameters().isEmpty())
162                            printError(methodElement,
163                                    "No parameters supported on @XMLFuzzy.Field method");
164                        final Map<String, Object> methodModel = new HashMap<>();
165                        methodModel.put("simpleName", methodElement.getSimpleName());
166                        final TypeMirror returnType = methodElement.getReturnType();
167                        if (VOID == returnType.getKind())
168                            printError(methodElement,
169                                    "Void return unsupported on @XMLFuzzy.Field method");
170                        methodModel.put("returnType", returnType);
171                        methodModel.put("xpath", methodElement.getAnnotation(Field.class).value());
172                        // TODO: Borrow from Converter to work out correct conversion method
173                        methodModel.put("converter", converterFor(methodElement, returnType));
174                        methodModel.put("nullable",
175                                0 == methodElement.getAnnotationsByType(Nonnull.class).length
176                                        && !returnType.getKind().isPrimitive());
177
178                        methodModels.add(methodModel);
179                    }
180                    model.put("methods", methodModels);
181
182                    configuration.getTemplate("fuzzy.ftl").process(model, out);
183                }
184            } catch (final IOException | TemplateException e) {
185                printError(element, e.toString());
186                return false;
187            }
188        }
189        return true;
190    }
191
192    private String converterFor(final ExecutableElement methodElement,
193            final TypeMirror returnType) {
194        try {
195            final Types typeUtils = processingEnv.getTypeUtils();
196            return typeUtils.boxedClass(typeUtils.getPrimitiveType(returnType.getKind()))
197                    + ".valueOf";
198        } catch (final IllegalArgumentException e) {
199            // Not primitive
200        }
201        try {
202            final Class<?> clazz = Class.forName(returnType.toString());
203            for (final String methodName : asList("parse", "valueOf", "of"))
204                for (final Class<?> parameterType : asList(String.class, CharSequence.class))
205                    try {
206                        if (isStatic(clazz.getMethod(methodName, parameterType).getModifiers()))
207                            return returnType + "." + methodName;
208                    } catch (final NoSuchMethodException ignored) {
209                    }
210            for (final Class<?> parameterType : asList(String.class, CharSequence.class))
211                try {
212                    clazz.getConstructor(parameterType);
213                    return "new " + returnType;
214                } catch (final NoSuchMethodException ignored) {
215                }
216        } catch (final ClassNotFoundException x) {
217            printError(methodElement, x.toString());
218        }
219        printError(methodElement, "No converter for '%s'", returnType);
220        return null;
221    }
222
223    private void printError(final Element element, final String format, final Object... args) {
224        processingEnv.getMessager()
225                .printMessage(ERROR, 0 == args.length ? format : format(format, args), element);
226    }
