001package hm.binkley.xml.reflect;
002
003import hm.binkley.util.Converter;
004import org.w3c.dom.Node;
005
006import javax.annotation.Nonnull;
007import javax.xml.xpath.XPath;
008import javax.xml.xpath.XPathExpression;
009import javax.xml.xpath.XPathExpressionException;
010import javax.xml.xpath.XPathFactory;
011import java.lang.reflect.InvocationHandler;
012import java.lang.reflect.Method;
013import java.util.Map;
014import java.util.concurrent.ConcurrentHashMap;
015
016import static java.lang.String.format;
017import static java.lang.reflect.Proxy.newProxyInstance;
018import static java.util.Arrays.asList;
019
020/**
021 * {@code RedoXMLBeam} <b>needs documentation</b>.
022 *
023 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
024 * @todo Needs documentation.
025 * @todo Replace run-time conversion with annotation processor to fix at compile time
026 * @see <a href="http://xml.org/">XMLBeam</a>
027 * @see <a href="http://en.wikipedia.org/wiki/Little_Fuzzy"><cite>Little Fuzzy</cite></a>
028 */
029public final class XMLFuzzy
030        implements InvocationHandler {
031    private static final XPath xpath = XPathFactory.newInstance().newXPath();
032    private static final Map<Method, XPathExpression> expressions = new ConcurrentHashMap<>();
033
034    private final Node node;
035    private final Converter converter;
036
037    public static final class Factory {
038        private final Converter converter;
039
040        public Factory(final Converter converter) {
041            this.converter = converter;
042        }
043
044        public <T> T of(@Nonnull final Class<T> itf, @Nonnull final Node node) {
045            return XMLFuzzy.of(itf, node, converter);
046        }
047    }
048
049    public static <T> T of(@Nonnull final Class<T> itf, @Nonnull final Node node,
050            @Nonnull final Converter converter) {
051        return itf.cast(newProxyInstance(itf.getClassLoader(), new Class[]{itf},
052                new XMLFuzzy(node, converter)));
053    }
054
055    private XMLFuzzy(final Node node, final Converter converter) {
056        this.node = node;
057        this.converter = converter;
058    }
059
060    @Override
061    public Object invoke(final Object proxy, final Method method, final Object[] args)
062            throws Throwable {
063        return converter.convert(method.getReturnType(), expressions.
064                computeIfAbsent(method, XMLFuzzy::compile).
065                evaluate(node));
066    }
067
068    private static XPathExpression compile(@Nonnull final Method method) {
069        final String expression = asList(method.getAnnotations()).stream().
070                filter(From.class::isInstance).
071                map(From.class::cast).
072                findFirst().
073                orElseThrow(() -> new MissingAnnotation(method)).
074                value();
075        try {
076            return xpath.compile(expression);
077        } catch (final XPathExpressionException e) {
078            throw new BadXPath(method, expression, e);
079        }
080    }
081
082    public static final class MissingAnnotation
083            extends RuntimeException {
084        private MissingAnnotation(final Method method) {
085            super(format("Missing @XMLFuzzy(xpath) annotation: %s", method));
086        }
087    }
088
089    public static final class BadXPath
090            extends RuntimeException {
091        private BadXPath(final Method method, final String expression,
092                final XPathExpressionException e) {
093            super(format("Bad @XMLFuzzy(xpath) annotation on '%s': %s: %s", method, expression,
094                    e.getMessage()));
095            setStackTrace(e.getStackTrace());
096        }
097    }
098}
