164@Documented
165@Inherited
166@Retention(SOURCE)
167@Target(TYPE)
168public @interface XMLFuzzy {
169    /**
170     * Marks interface methods with XPath expressions to extract method values from an XML
171     * node.Assigns missing values {@code null} unless marked with {@link Nonnull} in which case
172     * they throw {@code NullPointerException}.
173     * <p>
174     * Primitive return values are always treated as if marked with {@code
175     * javax.annotation.Nonnull}.  Void returns cause compilation errors.
176     */
177    @Documented
178    @Inherited
179    @Retention(SOURCE)
180    @Target(METHOD)
181    @interface Field {
182        @Language("XPath") String value();
183    }
184}
