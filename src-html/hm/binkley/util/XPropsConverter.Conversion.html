074    public XPropsConverter() {
075        register("address", value -> {
076            final HostAndPort parsed = HostAndPort.fromString(value)
077                    .requireBracketsForIPv6();
078            return createUnresolved(parsed.getHostText(), parsed.getPort());
079        });
080        register("bundle", ResourceBundle::getBundle);
081        register("byte", Byte::valueOf);
082        register("class", Class::forName);
083        register("date", LocalDate::parse);
084        register("decimal", BigDecimal::new);
085        register("double", Double::valueOf);
086        register("duration", Duration::parse);
087        register("file", File::new);
088        register("float", Float::valueOf);
089        register("inet", InetAddress::getByName);
090        register("int", Integer::valueOf);
091        register("integer", BigInteger::new);
092        register("long", Long::valueOf);
093        register("path", Paths::get);
094        register("period", Period::parse);
095        register("regex", Pattern::compile);
096        register("resource", value -> new DefaultResourceLoader(
097                currentThread().getContextClassLoader()).getResource(value));
098        register("resource*", value -> asList(
099                new PathMatchingResourcePatternResolver(
100                        currentThread().getContextClassLoader())
101                        .getResources(value)));
102        register("short", Short::valueOf);
103        register("time", LocalDateTime::parse);
104        register("timestamp", Instant::parse);
105        register("tz", TimeZone::getTimeZone);
106        register("uri", URI::create);
107        register("url", URL::new);
108    }

120    /**
121     * Registers a new alias mapping for converting property values to typed
122     * objects using the given <var>prefix</var> and <var>factory</var>.
123     * Factories <strong>should</strong> never return {@code null} values
124     * <p>
125     * The <var>type</var> prefix represents a Java or JDK type as listed here:
126     * <table><tr><th>Prefix</th> <th>Type</th></tr> <tr><td>address</td>
127     * <td>{@code java.net.InetSocketAddress}</td></tr> <tr><td>bundle</td>
128     * <td>{@code java.util.ResourceBundle}</td></tr> <tr><td>byte</td>
129     * <td>{@code java.lang.Byte}</td></tr> <tr><td>class</td>
130     * <td>java.lang.Class</td></tr> <tr><td>date</td> <td>{@code
131     * java.time.LocalDate}</td></tr> <tr><td>decimal</td> <td>{@code java.math
132     * .BigDecimal}</td></tr> <tr><td>double</td> <td>{@code
133     * java.lang.Double}</td></tr> <tr><td>duration</td> <td>{@code
134     * java.time.Duration}</td></tr> <tr><td>file</td> <td>{@code
135     * java.io.File}</td></tr> <tr><td>float</td> <td>{@code
136     * java.lang.Float}</td></tr> <tr><td>inet</td> <td>{@code
137     * java.net.InetAddress}</td></tr> <tr><td>int</td> <td>{@code
138     * java.lang.Integer}</td></tr> <tr><td>integer</td> <td>{@code
139     * java.math.BigInteger}</td></tr> <tr><td>long</td> <td>{@code
140     * java.lang.Long}</td></tr> <tr><td>path</td> <td>{@code
141     * java.nio.file.Path}</td></tr> <tr><td>period</td> <td>{@code
142     * java.time.Period}</td></tr> <tr><td>resource</td> <tr><td>regex</td>
143     * <td>{@code java.util.regex.Pattern}</td></tr> <td>{@code
144     * org.springframework.core.io.Resource}</td></tr> <tr><td>resource*</td>
145     * <td>{@code java.util.List&lt;org.springframework.core.io.Resource&gt;
146     * }</td></tr> <tr><td>short</td> <td>{@code java.lang.Short}</td></tr>
147     * <tr><td>time</td> <td>{@code java.time.LocalDateTime}</td></tr>
148     * <tr><td>timestamp</td> <td>{@code java.time.Instant}</td></tr>
149     * <tr><td>tz</td> <td>{@code java.util.TimeZone}</td></tr> <tr><td>uri</td>
150     * <td>{@code java.net.URI}</td></tr> <tr><td>url</td> <td>{@code
151     * java.net.URL}</td></tr> </table>
152     *
153     * @param prefix the alias prefix, never missing
154     * @param factory the factory, never missing
155     *
156     * @throws DuplicateConversionException if <var>prefix</var> is already
157     * registered
158     */
159    public void register(@Nonnull final String prefix,
160            @Nonnull final Conversion<?, ?> factory)
161            throws DuplicateConversionException {
162        if (null != factories.putIfAbsent(prefix, factory))
163            throw new DuplicateConversionException(prefix);
164    }

166    /**
167     * Converts the given <var>value</var> by <var>key</var>.
168     *
169     * @param key the conversion key, never missing
170     * @param value the value
171     * @param <T> the converted type, never missing
172     *
173     * @return the converted value
174     *
175     * @throws Exception if conversion fails
176     */
177    @SuppressWarnings("unchecked")
178    @Nonnull
179    public <T> T convert(@Nonnull final String key, @Nonnull final String value)
180            throws Exception {
181        return (T) factoryFor(key).convert(value);
182    }
183
184    @Nonnull
185    @SuppressWarnings({"unchecked", "ReuseOfLocalVariable"})
186    private <T, E extends Exception> Conversion<T, E> factoryFor(
187            final String type) {
188        // Look for alias first
189        Conversion<T, E> factory = (Conversion<T, E>) factories.get(type);
190        if (null != factory)
191            return factory;
192        final Class<T> token = tokenFor(type);
193        // Try Type.valueOf
194        factory = invokeValueOf(token);
195        if (null != factory)
196            return factory;
197        // Try Type.of
198        factory = invokeOf(token);
199        if (null != factory)
200            return factory;
201        // Try new Type
202        factory = invokeConstructor(token);
203        if (null != factory)
204            return factory;
205        throw new Bug("Unsupported conversion: %s", type);
206    }
207
208    @SuppressWarnings("unchecked")
209    private static <T, E extends Exception> Conversion<T, E> invokeValueOf(
210            final Class<T> token)
211            throws NoSuchMethodError {
212        try {
213            final Method method = token.getMethod("valueOf", String.class);
214            return value -> {
215                try {
216                    return (T) method.invoke(null, value);
217                } catch (final IllegalAccessException e) {
218                    final IllegalAccessError x = new IllegalAccessError(
219                            e.getMessage());
220                    x.setStackTrace(e.getStackTrace());
221                    throw x;
222                } catch (final InvocationTargetException e) {
223                    final Throwable root = getRootCause(e);
224                    final RuntimeException x = new RuntimeException(root);
225                    x.setStackTrace(root.getStackTrace());
226                    throw x;
227                }
228            };
229        } catch (final NoSuchMethodException ignored) {
230            return null;
231        }
232    }
233
234    @SuppressWarnings("unchecked")
235    private static <T, E extends Exception> Conversion<T, E> invokeOf(
236            final Class<T> token)
237            throws NoSuchMethodError {
238        try {
239            final Method method = token.getMethod("of", String.class);
240            return value -> {
241                try {
242                    return (T) method.invoke(null, value);
243                } catch (final IllegalAccessException e) {
244                    final IllegalAccessError x = new IllegalAccessError(
245                            e.getMessage());
246                    x.setStackTrace(e.getStackTrace());
247                    throw x;
248                } catch (final InvocationTargetException e) {
249                    final Throwable root = getRootCause(e);
250                    final RuntimeException x = new RuntimeException(root);
251                    x.setStackTrace(root.getStackTrace());
252                    throw x;
253                }
254            };
255        } catch (final NoSuchMethodException ignored) {
256            return null;
257        }
258    }
259
260    private static <T, E extends Exception> Conversion<T, E> invokeConstructor(
261            final Class<T> token)
262            throws NoSuchMethodError {
263        try {
264            final Constructor<T> ctor = token.getConstructor(String.class);
265            return value -> {
266                try {
267                    return ctor.newInstance(value);
268                } catch (final IllegalAccessException e) {
269                    final IllegalAccessError x = new IllegalAccessError(
270                            e.getMessage());
271                    x.setStackTrace(e.getStackTrace());
272                    throw x;
273                } catch (final InvocationTargetException e) {
274                    final Throwable root = getRootCause(e);
275                    final RuntimeException x = new RuntimeException(root);
276                    x.setStackTrace(root.getStackTrace());
277                    throw x;
278                } catch (final InstantiationException e) {
279                    final InstantiationError x = new InstantiationError(
280                            e.getMessage());
281                    x.setStackTrace(e.getStackTrace());
282                    throw x;
283                }
284            };
285        } catch (final NoSuchMethodException ignored) {
286            return null;
287        }
288    }
289
290    @SuppressWarnings("unchecked")
291    private static <T> Class<T> tokenFor(final String type) {
292        try {
293            return (Class<T>) Class.forName(type);
294        } catch (final ClassNotFoundException e) {
295            final IllegalArgumentException x = new IllegalArgumentException(
296                    type + ": " + e);
297            x.setStackTrace(e.getStackTrace());
298            throw x;
299        }
300    }
301
302    /**
303     * Converts a string property value into a typed object.
304     *
305     * @param <T> the converted type
306     * @param <E> the exception type on failed converstion, {@code
307     * RuntimeException} if none
308     */
309    @FunctionalInterface
310    public interface Conversion<T, E extends Exception> {
311        /**
312         * Converts the given property <var>value</var> into a typed object.
313         *
314         * @param value the property value, never missing
315         *
316         * @return the typed object
317         *
318         * @throws E if conversion fails
319         */
320        T convert(@Nonnull final String value)
321                throws E;
322    }
323
324    /** Duplicate key registered with {@link XPropsConverter}. */
325    public static class DuplicateConversionException
326            extends IllegalArgumentException {
327        /**
328         * Constructs a new {@code DuplicateConversionException} for the given
329         * <var>key</var>.
330         *
331         * @param key the conversion key, never missing
332         */
333        public DuplicateConversionException(@Nonnull final String key) {
334            super(key);
335        }
336    }
