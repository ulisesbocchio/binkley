107    /**
108     * Registers a new alias mapping for converting property values to typed objects using the given
109     * <var>prefix</var> and <var>factory</var>.
110     * <p>
111     * The <var>type</var> prefix represents a Java or JDK type as listed here:
112     * <table><tr><th>Prefix</th> <th>Type</th></tr> <tr><td>address</td> <td>{@code
113     * java.net.InetSocketAddress}</td></tr> <tr><td>bundle</td> <td>{@code
114     * java.util.ResourceBundle}</td></tr> <tr><td>byte</td> <td>{@code java.lang.Byte}</td></tr>
115     * <tr><td>class</td> <td>java.lang.Class</td></tr> <tr><td>date</td> <td>{@code
116     * java.time.LocalDate}</td></tr> <tr><td>decimal</td> <td>{@code java.math
117     * .BigDecimal}</td></tr>
118     * <tr><td>double</td> <td>{@code java.lang.Double}</td></tr> <tr><td>duration</td> <td>{@code
119     * java.time.Duration}</td></tr> <tr><td>file</td> <td>{@code java.io.File}</td></tr>
120     * <tr><td>float</td> <td>{@code java.lang.Float}</td></tr> <tr><td>inet</td> <td>{@code
121     * java.net.InetAddress}</td></tr> <tr><td>int</td> <td>{@code java.lang.Integer}</td></tr>
122     * <tr><td>integer</td> <td>{@code java.math.BigInteger}</td></tr> <tr><td>long</td> <td>{@code
123     * java.lang.Long}</td></tr> <tr><td>path</td> <td>{@code java.nio.file.Path}</td></tr>

126     * org.springframework.core.io.Resource}</td></tr> <tr><td>resource*</td> <td>{@code
127     * java.util.List&lt;org.springframework.core.io.Resource&gt;}</td></tr> <tr><td>short</td>
128     * <td>{@code java.lang.Short}</td></tr> <tr><td>time</td> <td>{@code
129     * java.time.LocalDateTime}</td></tr> <tr><td>timestamp</td> <td>{@code
130     * java.time.Instant}</td></tr> <tr><td>tz</td> <td>{@code java.util.TimeZone}</td></tr>
131     * <tr><td>uri</td> <td>{@code java.net.URI}</td></tr> <tr><td>url</td> <td>{@code
132     * java.net.URL}</td></tr> </table>
133     *
134     * @param prefix the alias prefix, never missing
135     * @param factory the factory, never missing
136     *
137     * @throws DuplicateConversionException if <var>prefix</var> is already registered
138     */
139    public void register(@Nonnull final String prefix, @Nonnull final Conversion<?, ?> factory)
140            throws DuplicateConversionException {
141        if (null != factories.putIfAbsent(prefix, factory))
142            throw new DuplicateConversionException(prefix);
143    }

145    /** @todo Documentation */
146    @SuppressWarnings("unchecked")
147    public <T> T convert(@Nonnull final String key, @Nonnull final String value)
148            throws Exception {
149        return (T) factoryFor(key).convert(value);
150    }
151
152    @Nonnull
153    @SuppressWarnings({"unchecked", "ReuseOfLocalVariable"})
154    private <T, E extends Exception> Conversion<T, E> factoryFor(final String type) {
155        // Look for alias first
156        Conversion<T, E> factory = (Conversion<T, E>) factories.get(type);
157        if (null != factory)
158            return factory;
159        final Class<T> token = tokenFor(type);
160        // Try Type.valueOf
161        factory = invokeValueOf(token);
162        if (null != factory)
163            return factory;
164        // Try Type.of
165        factory = invokeOf(token);
166        if (null != factory)
167            return factory;
168        // Try new Type
169        factory = invokeConstructor(token);
170        if (null != factory)
171            return factory;
172        throw new Bug("Unsupported conversion: %s", type);
173    }
174
175    @SuppressWarnings("unchecked")
176    private static <T, E extends Exception> Conversion<T, E> invokeValueOf(final Class<T> token)
177            throws NoSuchMethodError {
178        try {
179            final Method method = token.getMethod("valueOf", String.class);
180            return value -> {
181                try {
182                    return (T) method.invoke(null, value);
183                } catch (final IllegalAccessException e) {
184                    final IllegalAccessError x = new IllegalAccessError(e.getMessage());
185                    x.setStackTrace(e.getStackTrace());
186                    throw x;
187                } catch (final InvocationTargetException e) {
188                    final Throwable root = getRootCause(e);
189                    final RuntimeException x = new RuntimeException(root);
190                    x.setStackTrace(root.getStackTrace());
191                    throw x;
192                }
193            };
194        } catch (final NoSuchMethodException ignored) {
195            return null;
196        }
197    }
198
199    @SuppressWarnings("unchecked")
200    private static <T, E extends Exception> Conversion<T, E> invokeOf(final Class<T> token)
201            throws NoSuchMethodError {
202        try {
203            final Method method = token.getMethod("of", String.class);
204            return value -> {
205                try {
206                    return (T) method.invoke(null, value);
207                } catch (final IllegalAccessException e) {
208                    final IllegalAccessError x = new IllegalAccessError(e.getMessage());
209                    x.setStackTrace(e.getStackTrace());
210                    throw x;
211                } catch (final InvocationTargetException e) {
212                    final Throwable root = getRootCause(e);
213                    final RuntimeException x = new RuntimeException(root);
214                    x.setStackTrace(root.getStackTrace());
215                    throw x;
216                }
217            };
218        } catch (final NoSuchMethodException ignored) {
219            return null;
220        }
221    }
222
223    private static <T, E extends Exception> Conversion<T, E> invokeConstructor(final Class<T> token)
224            throws NoSuchMethodError {
225        try {
226            final Constructor<T> ctor = token.getConstructor(String.class);
227            return value -> {
228                try {
229                    return ctor.newInstance(value);
230                } catch (final IllegalAccessException e) {
231                    final IllegalAccessError x = new IllegalAccessError(e.getMessage());
232                    x.setStackTrace(e.getStackTrace());
233                    throw x;
234                } catch (final InvocationTargetException e) {
235                    final Throwable root = getRootCause(e);
236                    final RuntimeException x = new RuntimeException(root);
237                    x.setStackTrace(root.getStackTrace());
238                    throw x;
239                } catch (final InstantiationException e) {
240                    final InstantiationError x = new InstantiationError(e.getMessage());
241                    x.setStackTrace(e.getStackTrace());
242                    throw x;
243                }
244            };
245        } catch (final NoSuchMethodException ignored) {
246            return null;
247        }
248    }
249
250    @SuppressWarnings("unchecked")
251    private static <T> Class<T> tokenFor(final String type) {
252        try {
253            return (Class<T>) Class.forName(type);
254        } catch (final ClassNotFoundException e) {
255            final IllegalArgumentException x = new IllegalArgumentException(type + ": " + e);
256            x.setStackTrace(e.getStackTrace());
257            throw x;
258        }
259    }
260
261    /**
262     * Converts a string property value into a typed object.
263     *
264     * @param <T> the converted type
265     * @param <E> the exception type on failed converstion, {@code RuntimeException} if none
266     */
267    @FunctionalInterface
268    public interface Conversion<T, E extends Exception> {
269        /**
270         * Converts the given property <var>value</var> into a typed object.
271         *
272         * @param value the property value, never missing
273         *
274         * @return the typed object
275         *
276         * @throws E if conversion fails
277         */
278        T convert(@Nonnull final String value)
279                throws E;
280    }
