007package hm.binkley.util;
008
009import javax.annotation.Nonnull;
010import java.util.List;
011
012import static java.lang.reflect.Proxy.newProxyInstance;
013
014/**
015 * {@code Mixin} implements concrete class mixins via JDK proxies.  Proxy
016 * handler strategy tries: <ol><li>Previously matched methods</li> <li>Exact
017 * static match by enclosing type</li> <li>"Duck" matches by name and parameter
018 * type</li></ol>  Mixins expose a single, <em>composite interface</em> of all
019 * supported interfaces.
020 * <p>
021 * Mixins may optionally include {@code Mixin} as port of the <em>composite
022 * interface</em>.  This exposes {@link #mixinDelegates()} providing public
023 * access to the mixed in object delegated to by this proxy in the same order as
024 * method lookup considers delegates.
025 *
026 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
027 * @see #newMixin(Class, Object...) Create a new mixin
028 */
029public interface Mixin {
030    /**
031     * Creates a new mixin.  If <var>as</var> extends {@code Mixin}, provides a
032     * supporting {@link #mixinDelegates()} method giving public access to
033     * <var>delegates</var>.
034     *
035     * @param as a superinterface of visible public methods implemented by
036     * <var>delegates</var>, never missing
037     * @param delegates the mixed-in delegates (implementations)
038     * @param <T> the mixin proxy type (type of <var>as</var>)
039     *
040     * @return the mixed-in proxy, never missing
041     */
042    @Nonnull
043    static <T> T newMixin(@Nonnull final Class<T> as,
044            final Object... delegates) {
045        return as.cast(newProxyInstance(as.getClassLoader(), new Class[]{as},
046                new MixinHandler<>(as,
047                        new MixedDelegates(delegates).mixinDelegates())));
048    }
049
050    /**
051     * Gets an unmodifiable list of delegates implementing the mixin in the same
052     * order as method lookup.
053     *
054     * @return the unmodifiable list of delegates, never missing
055     */
056    @Nonnull
057    List<Object> mixinDelegates();
058}
