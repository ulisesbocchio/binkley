007package hm.binkley.util;
008
009import javax.annotation.Nonnull;
010import java.lang.invoke.MethodHandle;
011import java.lang.invoke.MethodHandles;
012import java.lang.invoke.MethodHandles.Lookup;
013import java.lang.reflect.InvocationHandler;
014import java.lang.reflect.Method;
015import java.util.ArrayList;
016import java.util.List;
017import java.util.concurrent.ConcurrentHashMap;
018import java.util.concurrent.ConcurrentMap;
019import java.util.function.Function;
020
021import static java.lang.String.format;
022import static java.lang.reflect.Proxy.newProxyInstance;
023import static java.util.Arrays.asList;
024import static java.util.Collections.unmodifiableList;
025
026/**
027 * {@code Mixin} implements concrete class mixins via JDK proxies.  Proxy handler strategy tries:
028 * <ol><li>Previously matched methods</li> <li>Exact static match by enclosing type</li> <li>"Duck"
029 * matches by name and parameter type</li></ol>  Mixins expose a single, <em>composite
030 * interface</em> of all supported interfaces.
031 * <p>
032 * Mixins may optionally include {@code Mixin} as port of the <em>composite interface</em>.  This
033 * exposes {@link #mixinDelegates()} providing public access to the mixed in object delegated to by
034 * this proxy.
035 *
036 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
037 */
038public interface Mixin {
039    /**
040     * Gets an unmodifiable list of delegates implementing the mixin.
041     *
042     * @return the unmodifiable list of delegates, never missing
043     */
044    @Nonnull
045    List<Object> mixinDelegates();
046
047    final class Factory {
048        /**
049         * Creates a new mixin.  If <var>as</var> extends {@code Mixin}, provides a supporting
050         * {@link #mixinDelegates()} method to give public access to <var>delegates</var>.
051         *
052         * @param as a superinterface of visible public methods implemented by <var>delegates</var>,
053         * never missing
054         * @param delegates the mixed-in delegates (implementations)
055         * @param <T> the mixin proxy type (type of <var>as</var>)
056         *
057         * @return the mixed-in proxy, never missing
058         */
059        @Nonnull
060        public static <T> T newMixin(@Nonnull final Class<T> as, final Object... delegates) {
061            return as.cast(newProxyInstance(as.getClassLoader(), new Class[]{as},
062                    new Handler<>(as, new MixedDelegates(delegates).mixinDelegates())));
063        }
064
065        private Factory() {
066        }
067
068        private static class Handler<T>
069                implements InvocationHandler {
070            private static final Lookup LOOKUP = MethodHandles.lookup();
071            private final ConcurrentMap<Method, MethodHandle> matches;
072            private final List<Object> delegates;
073
074            private Handler(final Class<T> as, final List<Object> delegates) {
075                this.delegates = delegates(as, delegates);
076                matches = new ConcurrentHashMap<>(as.getMethods().length);
077            }
078
079            private List<Object> delegates(final Class<T> as, final List<Object> delegates) {
080                for (final Method method : as.getMethods())
081                    if (method.isDefault())
082                        try {
083                            final List<Object> delegatesPlus = new ArrayList<>(
084                                    delegates.size() + 1);
085                            delegatesPlus.addAll(delegates);
086                            delegatesPlus.add(InterfaceInstance.newInstance(as));
087                            return delegatesPlus;
088                        } catch (final IllegalAccessException e) {
089                            throw copy(e, IllegalAccessError::new);
090                        } catch (final ClassNotFoundException e) {
091                            throw copy(e, UnknownError::new);
092                        } catch (final InstantiationException e) {
093                            throw copy(e, InstantiationError::new);
094                        }
095                return delegates;
096            }
097
098            @Override
099            public Object invoke(@Nonnull final Object proxy, @Nonnull final Method method,
100                    final Object[] args)
101                    throws Throwable {
102                // Try as previous found - method match memoized
103                final MethodHandle handle = matches.get(method);
104                if (null != handle)
105                    return handle.invoke(args);
106                final MethodHandle unreflected = LOOKUP.unreflect(method);
107                // Try as an implementation - static typing
108                for (final Object delegate : delegates)
109                    try {
110                        final MethodHandle bound = LOOKUP
111                                .findVirtual(delegate.getClass(), method.getName(),
112                                        unreflected.type()).bindTo(delegate);
113                        final Object value = bound.invokeWithArguments(args);
114                        matches.put(method, bound);
115                        return value;
116                    } catch (final NoSuchMethodException e) {
117                        // Try next method
118                    }
119                // Try by name/arg types - duck typing
120                final String name = method.getName();
121                final Class<?>[] types = method.getParameterTypes();
122                for (final Object delegate : delegates) {
123                    final Class<?> duck = delegate.getClass();
124                    try {
125                        final Method quack = duck.getMethod(name, types);
126                        if (!method.getReturnType().isAssignableFrom(quack.getReturnType()))
127                            continue;
128                        final MethodHandle bound = LOOKUP.unreflect(quack).bindTo(delegate);
129                        final Object value = bound.invokeWithArguments(args);
130                        matches.put(method, bound);
131                        return value;
132                    } catch (final NoSuchMethodException ignored) {
133                        // Try next delegate
134                    }
135                }
136                throw new AbstractMethodError(
137                        format("BUG: Missing implementation for <%s> among %s.", method,
138                                delegates));
139            }
140        }
141
142        private static <E extends Throwable> E copy(final Throwable from,
143                final Function<String, E> ctor) {
144            final E to = ctor.apply(from.getMessage());
145            to.setStackTrace(from.getStackTrace());
146            for (final Throwable s : from.getSuppressed())
147                to.addSuppressed(s);
148            return to;
149        }
150
151        private static class MixedDelegates
152                implements Mixin {
153            private final List<Object> mixed;
154
155            public MixedDelegates(final Object... delegates) {
156                final List<Object> mixed = new ArrayList<>(delegates.length + 1);
157                mixed.addAll(asList(delegates));
158                mixed.add(this);
159                this.mixed = unmodifiableList(mixed);
160            }
161
162            @SuppressWarnings("ReturnOfCollectionOrArrayField")
163            @Nonnull
164            @Override
165            public List<Object> mixinDelegates() {
166                return mixed;
167            }
168        }
169    }
