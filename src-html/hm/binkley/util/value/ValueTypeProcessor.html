028package hm.binkley.util.value;
029
030import org.kohsuke.MetaInfServices;
031
032import javax.annotation.processing.AbstractProcessor;
033import javax.annotation.processing.Processor;
034import javax.annotation.processing.RoundEnvironment;
035import javax.annotation.processing.SupportedAnnotationTypes;
036import javax.annotation.processing.SupportedSourceVersion;
037import javax.lang.model.element.AnnotationMirror;
038import javax.lang.model.element.AnnotationValue;
039import javax.lang.model.element.Element;
040import javax.lang.model.element.ExecutableElement;
041import javax.lang.model.element.TypeElement;
042import javax.lang.model.type.DeclaredType;
043import javax.tools.Diagnostic.Kind;
044import java.io.IOException;
045import java.io.OutputStream;
046import java.util.List;
047import java.util.Scanner;
048import java.util.Set;
049import java.util.regex.Pattern;
050
051import static java.lang.String.format;
052import static java.nio.charset.StandardCharsets.UTF_8;
053import static java.util.regex.Pattern.compile;
054import static javax.lang.model.SourceVersion.RELEASE_8;
055import static javax.lang.model.element.ElementKind.INTERFACE;
056import static javax.lang.model.element.ElementKind.PACKAGE;
057import static javax.lang.model.element.Modifier.STATIC;
058import static javax.tools.Diagnostic.Kind.ERROR;
059import static javax.tools.Diagnostic.Kind.NOTE;
060import static javax.tools.Diagnostic.Kind.WARNING;
061
062/**
063 * {@code ValueTypeProcessor} <b>needs documentation</b>.
064 *
065 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
066 * @todo Needs documentation.
067 */
068@MetaInfServices(Processor.class)
069@SupportedAnnotationTypes("hm.binkley.util.value.ValueType")
070@SupportedSourceVersion(RELEASE_8)
071public final class ValueTypeProcessor
072        extends AbstractProcessor {
073    private static final Pattern packageVar = compile("\\$\\{package\\}");
074    private static final Pattern classVar = compile("\\$\\{class\\}");
075    private static final Pattern baseVar = compile("\\$\\{base\\}");
076    private static final Pattern typeVar = compile("\\$\\{type\\}");
077    private static final Pattern modifyVar = compile("\\$\\{modify\\}");
078
079    private final String template;
080
081    public ValueTypeProcessor() {
082        try (final Scanner scanner = new Scanner(getClass().getResourceAsStream("value-type.java"),
083                UTF_8.name())) {
084            template = scanner.useDelimiter("\\A").next();
085        }
086    }
087
088    @Override
089    public boolean process(final Set<? extends TypeElement> annotations,
090            final RoundEnvironment roundEnv) {
091        ELEMENT:
092        for (final Element element : roundEnv.getElementsAnnotatedWith(
093                processingEnv.getElementUtils().getTypeElement(ValueType.class.getName()))) {
094            final List<? extends AnnotationMirror> aMirrors = element.getAnnotationMirrors();
095            for (final AnnotationMirror mirror : aMirrors) {
096                if (!ValueType.class.getName().equals(mirror.getAnnotationType().toString()))
097                    continue;
098
099                final Messenger messenger = new Messenger(element, mirror);
100
101                if (INTERFACE != element.getKind()) {
102                    messenger.error("@ValueType only supported on interfaces");
103                    continue;
104                }
105
106                if (PACKAGE != element.getEnclosingElement().getKind()) {
107                    messenger.error("@ValueType only supported on top-level interfaces");
108                    continue;
109                }
110
111                ((TypeElement) element).getEnclosedElements().stream().
112                        map(ExecutableElement.class::cast).
113                        filter(ValueTypeProcessor::unsupported).
114                        forEach(m -> messenger
115                                .error("@ValueType supports only static and default methods: %s",
116                                        m));
117
118                // Rely on knowledge there is only the value field
119                final AnnotationValue value = mirror.getElementValues().values().stream().
120                        findFirst().get();
121
122                final String packaj = processingEnv.getElementUtils().getPackageOf(element)
123                        .getQualifiedName().toString();
124                final String clazz = element.getSimpleName().toString();
125                final String source = format("%s.%sValue", packaj, clazz);
126                final DeclaredType valueType = (DeclaredType) value.getValue();
127                final String type = valueType.toString();
128                final String modify = "java.lang.String".equals(type) ? ".intern()" : "";
129
130                final boolean comparable = comparable(valueType);
131                final String base = comparable ? "ComparableValue" : "Value";
132
133                try (final OutputStream out = processingEnv.getFiler()
134                        .createSourceFile(source, element).openOutputStream()) {
135                    String generated = packageVar.matcher(template).replaceAll(packaj);
136                    generated = classVar.matcher(generated).replaceAll(clazz);
137                    generated = baseVar.matcher(generated).replaceAll(base);
138                    generated = typeVar.matcher(generated).replaceAll(type);
139                    generated = modifyVar.matcher(generated).replaceAll(modify);
140                    out.write(generated.getBytes(UTF_8));
141                } catch (final IOException e) {
142                    messenger.error("Cannot generate source: %s", e);
143                    continue ELEMENT;
144                }
145
146                messenger.note("Generated %s.java", source);
147            }
148        }
149
150        return true;
151    }
152
153    private static boolean unsupported(final ExecutableElement e) {
154        return !(e.isDefault() || e.getModifiers().contains(STATIC));
155    }
156
157    private static boolean comparable(final DeclaredType type) {
158        final String comparable = format("java.lang.Comparable<%s>", type);
159        return ((TypeElement) type.asElement()).getInterfaces().stream().
160                map(Object::toString).
161                filter(comparable::equals).
162                findFirst().
163                isPresent();
164    }
165
166    private final class Messenger {
167        private final Element element;
168        private final AnnotationMirror mirror;
169
170        private Messenger(final Element element, final AnnotationMirror mirror) {
171            this.element = element;
172            this.mirror = mirror;
173        }
174
175        void error(final String format, final Object... args) {
176            message(ERROR, format, args);
177        }
178
179        void warning(final String format, final Object... args) {
180            message(WARNING, format, args);
181        }
182
183        void note(final String format, final Object... args) {
184            message(NOTE, format, args);
185        }
186
187        private void message(final Kind kind, final String format, final Object... args) {
188            processingEnv.getMessager().printMessage(kind, format(format, args), element, mirror);
189        }
190    }
191}
