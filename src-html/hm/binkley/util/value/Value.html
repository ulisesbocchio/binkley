001package hm.binkley.util.value;
002
003import lombok.EqualsAndHashCode;
004
005import javax.annotation.Nonnull;
006import java.util.function.Function;
007
008import static java.util.Objects.requireNonNull;
009
010/**
011 * {@code Value} <b>needs documentation</b>.
012 *
013 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
014 * @todo Needs documentation.
015 */
016@EqualsAndHashCode
017public abstract class Value<T, V extends Value<T, V>>
018        implements Cloneable {
019    @Nonnull
020    protected final T value;
021
022    protected Value(@Nonnull final T value) {
023        this.value = value;
024    }
025
026    @Nonnull
027    public abstract Class<T> type();
028
029    @Nonnull
030    public final T get() {
031        return value;
032    }
033
034    @Nonnull
035    public final <U, W extends Value<U, W>> W map(
036            @Nonnull final Function<? super T, ? extends W> mapper) {
037        requireNonNull(mapper, "mapper");
038        return mapper.apply(value);
039    }
040
041    @SuppressWarnings({"CloneDoesntCallSuperClone", "unchecked"})
042    @Override
043    public final V clone() {
044        return (V) this;
045    }
046
047    @Override
048    public final String toString() {
049        return "<" + value + ">";
050    }
051}
