001package hm.binkley.util.value;
002
003import javax.annotation.Nonnull;
004
005import static java.util.Objects.requireNonNull;
006
007/**
008 * {@code StringValue} <b>needs documentation</b>.
009 *
010 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
011 * @todo Needs documentation.
012 */
013public abstract class ComparableValue<T extends Comparable<T>, V extends ComparableValue<T, V>>
014        extends Value<T, V>
015        implements Comparable<V> {
016    protected ComparableValue(@Nonnull final T value) {
017        super(value);
018    }
019
020    @Override
021    public final int compareTo(@Nonnull final V that) {
022        requireNonNull(that, "that");
023        return value.compareTo(that.value);
024    }
025}
