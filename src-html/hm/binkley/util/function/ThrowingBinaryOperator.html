001package hm.binkley.util.function;
002
003import javax.annotation.Nonnull;
004import java.util.Comparator;
005import java.util.function.BinaryOperator;
006
007/**
008 * {@code ThrowingBinaryOperator} is a <em>throwing</em> look-a=like of {@link BinaryOperator}.  It
009 * cannot be a {@code BinaryOperator} as it takes throwing versions of binary operators.  Otherwise
010 * it is a faithful reproduction.
011 *
012 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
013 */
014@SuppressWarnings({"UnusedDeclaration", "JavaDoc"})
015@FunctionalInterface
016public interface ThrowingBinaryOperator<T, E extends Exception>
017        extends ThrowingBiFunction<T, T, T, E> {
018    /** @see BinaryOperator#minBy(Comparator) */
019    @Nonnull
020    static <T, E extends Exception> ThrowingBinaryOperator<T, E> minBy(
021            @Nonnull final Comparator<? super T> comparator) {
022        return (a, b) -> 0 >= comparator.compare(a, b) ? a : b;
023    }
024
025    /** @see BinaryOperator#maxBy(Comparator) */
026    @Nonnull
027    static <T, E extends Exception> ThrowingBinaryOperator<T, E> maxBy(
028            @Nonnull final Comparator<? super T> comparator) {
029        return (a, b) -> 0 <= comparator.compare(a, b) ? a : b;
030    }
031
032    /** Creates a facade {@code BinaryOperator} wrapping this throwing one. */
033    default <D extends RuntimeException> BinaryOperator<T> asBinaryOperator(final Defer<D> defer) {
034        return (u, v) -> defer.as(() -> apply(u, v));
035    }
036}
