001package hm.binkley.util.function;
002
003/**
004 * {@code ThrowingRunnable} is a <em>throwing</em> look-a=like of {@link Runnable}.  It cannot be a
005 * {@code Runnable} as it throws.  Otherwise it is a faithful reproduction.
006 *
007 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
008 */
009@SuppressWarnings("JavaDoc")
010@FunctionalInterface
011public interface ThrowingRunnable<E extends Exception> {
012    /** @see Runnable#run() */
013    void run()
014            throws E, InterruptedException;
015
016    /** Creates a facade {@code Runnable} wrapping this throwing one. */
017    default <D extends RuntimeException> Runnable asRunnable(final Defer<D> defer) {
018        return () -> defer.as(this);
019    }
020}
