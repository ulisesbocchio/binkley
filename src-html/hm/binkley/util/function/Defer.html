028package hm.binkley.util.function;
029
030import java.util.concurrent.CancellationException;
031import java.util.function.Function;
032
033import static java.lang.Thread.currentThread;
034
035/**
036 * {@code ExceptionTranslator} <b>needs documentation</b>.
037 *
038 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
039 * @todo Needs documentation.
040 * @todo Package scope or public?
041 */
042@FunctionalInterface
043public interface Defer<D extends RuntimeException>
044        extends Function<Exception, D> {
045    default <T, E extends Exception> T as(final ThrowingSupplier<? extends T, E> throwing) {
046        try {
047            return throwing.get();
048        } catch (final CancellationException e) {
049            throw apply(e);
050        } catch (final RuntimeException e) {
051            throw e;
052        } catch (final InterruptedException e) {
053            currentThread().interrupt();
054            throw apply(e);
055        } catch (final Exception e) {
056            throw apply(e);
057        }
058    }
059
060    default <E extends Exception> boolean as(final ThrowingBooleanSupplier<E> throwing) {
061        try {
062            return throwing.getAsBoolean();
063        } catch (final CancellationException e) {
064            throw apply(e);
065        } catch (final RuntimeException e) {
066            throw e;
067        } catch (final InterruptedException e) {
068            currentThread().interrupt();
069            throw apply(e);
070        } catch (final Exception e) {
071            throw apply(e);
072        }
073    }
074
075    default <E extends Exception> long as(final ThrowingLongSupplier<E> throwing) {
076        try {
077            return throwing.getAsLong();
078        } catch (final CancellationException e) {
079            throw apply(e);
080        } catch (final RuntimeException e) {
081            throw e;
082        } catch (final InterruptedException e) {
083            currentThread().interrupt();
084            throw apply(e);
085        } catch (final Exception e) {
086            throw apply(e);
087        }
088    }
089
090    default <E extends Exception> void as(final ThrowingRunnable<E> throwing) {
091        try {
092            throwing.run();
093        } catch (final CancellationException e) {
094            throw apply(e);
095        } catch (final RuntimeException e) {
096            throw e;
097        } catch (final InterruptedException e) {
098            currentThread().interrupt();
099            throw apply(e);
100        } catch (final Exception e) {
101            throw apply(e);
102        }
103    }
104}
