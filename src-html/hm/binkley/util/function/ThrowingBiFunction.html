001package hm.binkley.util.function;
002
003import javax.annotation.Nonnull;
004import java.util.function.BiFunction;
005import java.util.function.Function;
006
007/**
008 * {@code ThrowingBiFunction} is a <em>throwing</em> look-a=like of {@link BiFunction}.  It cannot
009 * be a {@code BiFunction} as it takes throwing versions of bi-functions.  Otherwise it is a
010 * faithful reproduction.
011 *
012 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
013 */
014@SuppressWarnings({"UnusedDeclaration", "JavaDoc"})
015@FunctionalInterface
016public interface ThrowingBiFunction<T, U, R, E extends Exception> {
017    /** @see BiFunction#apply(Object, Object) */
018    R apply(final T t, final U u)
019            throws E, InterruptedException;
020
021    /** @see BiFunction#andThen(Function) */
022    @Nonnull
023    default <V> ThrowingBiFunction<T, U, V, E> andThen(
024            @Nonnull final ThrowingFunction<? super R, ? extends V, E> after)
025            throws E, InterruptedException {
026        return (T t, U u) -> after.apply(apply(t, u));
027    }
028
029    /** Creates a facade {@code Function} wrapping this throwing one. */
030    default <D extends RuntimeException> BiFunction<T, U, R> asBiFunction(final Defer<D> defer) {
031        return (u, v) -> defer.as(() -> apply(u, v));
032    }
033}
