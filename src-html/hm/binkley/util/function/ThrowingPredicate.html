001package hm.binkley.util.function;
002
003import javax.annotation.Nonnull;
004import java.util.Objects;
005import java.util.function.Predicate;
006
007/**
008 * {@code ThrowingPredicate} is a <em>throwing</em> look-a=like of {@link Predicate}.  It cannot be
009 * a {@code Predicate} as it takes throwing versions of predicates.  Otherwise it is a faithful
010 * reproduction.
011 *
012 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
013 */
014@SuppressWarnings({"UnusedDeclaration", "JavaDoc"})
015@FunctionalInterface
016public interface ThrowingPredicate<T, E extends Exception> {
017    /** @see Predicate#test(Object) */
018    boolean test(final T t)
019            throws E, InterruptedException;
020
021    /** @see Predicate#and(Predicate) */
022    @Nonnull
023    default ThrowingPredicate<T, E> and(@Nonnull final ThrowingPredicate<? super T, E> other) {
024        return t -> test(t) && other.test(t);
025    }
026
027    /** @see Predicate#negate() */
028    @Nonnull
029    default ThrowingPredicate<T, E> negate() {
030        return t -> !test(t);
031    }
032
033    /** @see Predicate#or(Predicate) */
034    @Nonnull
035    default ThrowingPredicate<T, E> or(@Nonnull final ThrowingPredicate<? super T, E> other) {
036        return t -> test(t) || other.test(t);
037    }
038
039    /** @see Predicate#isEqual(Object) */
040    @Nonnull
041    static <T> ThrowingPredicate<T, RuntimeException> isEqual(
042            final Object targetRef) {
043        return null == targetRef ? Objects::isNull : targetRef::equals;
044    }
045
046    /** Creates a facade {@code Predicate} wrapping this throwing one. */
047    default <D extends RuntimeException> Predicate<T> asPredicate(
048            final Defer<D> defer) {
049        return t -> defer.as(() -> test(t));
050    }
051}
