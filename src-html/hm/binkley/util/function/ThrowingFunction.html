001package hm.binkley.util.function;
002
003import javax.annotation.Nonnull;
004import java.util.function.Function;
005
006/**
007 * {@code ThrowingFunction} is a <em>throwing</em> look-a=like of {@link Function}.  It cannot be a
008 * {@code Function} as it takes throwing versions of functions.  Otherwise it is a faithful
009 * reproduction.
010 *
011 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
012 */
013@SuppressWarnings({"UnusedDeclaration", "JavaDoc"})
014@FunctionalInterface
015public interface ThrowingFunction<T, R, E extends Exception> {
016    /** @see Function#apply(Object) */
017    R apply(final T t)
018            throws E, InterruptedException;
019
020    /** @see Function#compose(Function) */
021    @Nonnull
022    default <V> ThrowingFunction<V, R, E> compose(
023            @Nonnull final ThrowingFunction<? super V, ? extends T, E> before) {
024        return (V v) -> apply(before.apply(v));
025    }
026
027    /** @see Function#andThen(Function) */
028    @Nonnull
029    default <V> ThrowingFunction<T, V, E> andThen(
030            @Nonnull final ThrowingFunction<? super R, ? extends V, E> after) {
031        return (T t) -> after.apply(apply(t));
032    }
033
034    /** @see Function#identity() */
035    @Nonnull
036    static <T> ThrowingFunction<T, T, RuntimeException> identity() {
037        return t -> t;
038    }
039
040    /** Creates a facade {@code Function} wrapping this throwing one. */
041    default <D extends RuntimeException> Function<T, R> asFunction(
042            final Defer<D> defer) {
043        return t -> defer.as(() -> apply(t));
044    }
045}
