001package hm.binkley.util.function;
002
003import java.util.function.BooleanSupplier;
004
005/**
006 * {@code ThrowingBooleanSupplier} is a <em>throwing</em> look-a=like of {@link BooleanSupplier}. It
007 * cannot be a {@code BooleanSupplier} as it takes throwing versions of boolean suppliers. Otherwise
008 * it is a faithful reproduction.
009 *
010 * @author <a href="mailto:binkley@alumni.rice.edu">B. K. Oxley (binkley)</a>
011 */
012@SuppressWarnings("JavaDoc")
013@FunctionalInterface
014public interface ThrowingBooleanSupplier<E extends Exception> {
015    /** @see BooleanSupplier#getAsBoolean() */
016    boolean getAsBoolean()
017            throws E, InterruptedException;
018
019    /** Creates a facade {@code BooleanSupplier} wrapping this throwing one. */
020    default <D extends RuntimeException> BooleanSupplier asBooleanSupplier(final Defer<D> defer) {
021        return () -> defer.as(this);
022    }
023}
